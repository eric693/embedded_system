軟體分層架構原則

嵌入式專案常因「業務邏輯與硬體操作混雜」導致難以測試和移植。 標準做法是採用 HAL → Driver → Service → Application 四層架構， 每層只依賴下一層的介面，不跨層直接呼叫暫存器。

軟體分層架構 ┌─────────────────────────────────────────────┐ │ Application Layer main_task, cli_task │ │ 業務邏輯、狀態機、使用者互動 │ ├─────────────────────────────────────────────┤ │ Service Layer sensor_svc, mqtt_svc │ │ 協定封裝、資料聚合、錯誤重試 │ ├─────────────────────────────────────────────┤ │ Driver Layer bme280, enc_ab, pwm │ │ 晶片驅動、暫存器操作、中斷處理 │ ├─────────────────────────────────────────────┤ │ HAL Layer i2c_hal, uart_hal │ │ STM32 HAL / LL 或 Linux 系統呼叫 │ └─────────────────────────────────────────────┘

BME280 I²C 驅動實作

BME280 預設 I²C 位址為 0x76（SDO 接 GND）或 0x77（SDO 接 VCC）。 讀取補償參數後套用 Bosch 官方補償公式，得到校正後的溫度（°C）、濕度（%RH）與氣壓（hPa）。

C — bme280_driver.c（核心片段） 
#define BME280_ADDR   0x76
#define REG_CTRL_HUM  0xF2
#define REG_CTRL_MEAS 0xF4
#define REG_CONFIG    0xF5
#define REG_DATA      0xF7  /* 8 bytes: press(3)+temp(3)+hum(2) */

typedef struct {
    int32_t  temperature;   /* °C × 100，e.g. 2534 = 25.34°C */
    uint32_t humidity;      /* %RH × 1024 */
    uint32_t pressure;      /* Pa × 256 */
} bme280_data_t;

void bme280_init(void)
{
    /* 過取樣 ×1、Normal mode、standby 1000 ms */
    uint8_t buf[2];

    buf[0] = REG_CTRL_HUM; buf[1] = 0x01;
    i2c_write(BME280_ADDR, buf, 2);

    buf[0] = REG_CTRL_MEAS; buf[1] = 0x27;  /* temp ×1, press ×1, normal */
    i2c_write(BME280_ADDR, buf, 2);

    buf[0] = REG_CONFIG; buf[1] = 0xA0;   /* standby 1000 ms, filter ×4 */
    i2c_write(BME280_ADDR, buf, 2);

    bme280_read_calib();  /* 讀取補償係數（0x88–0x9F, 0xA1, 0xE1–0xF0）*/
}

int bme280_read(bme280_data_t *out)
{
    uint8_t raw[8];
    if (i2c_read_reg(BME280_ADDR, REG_DATA, raw, sizeof(raw)) != 0)
        return -1;

    int32_t adc_T = (raw[3] << 12) | (raw[4] << 4) | (raw[5] >> 4);
    int32_t adc_H = (raw[6] <<  8) |  raw[7];

    out->temperature = bme280_compensate_T(adc_T);   /* 設定 t_fine */
    out->humidity    = bme280_compensate_H(adc_H);
    out->pressure    = bme280_compensate_P(
        (raw[0] << 12) | (raw[1] << 4) | (raw[2] >> 4));
    return 0;
}
FreeRTOS 任務設計

系統分為三個任務：sensor_task（讀取感測器，放入 Queue）、 mqtt_task（從 Queue 取資料，組 JSON 後透過 UART 發送 AT 指令給 ESP8266）、 led_task（心跳燈，指示系統狀態）。

C — main.c（任務建立） 
#include "FreeRTOS.h"
#include "task.h"
#include "queue.h"

QueueHandle_t xSensorQueue;

void sensor_task(void *arg)
{
    bme280_data_t data;
    for (;;) {
        if (bme280_read(&data) == 0)
            xQueueSend(xSensorQueue, &data, 0);
        vTaskDelay(pdMS_TO_TICKS(10000));  /* 每 10 s 採樣一次 */
    }
}

void mqtt_task(void *arg)
{
    bme280_data_t data;
    char json[128];
    for (;;) {
        if (xQueueReceive(xSensorQueue, &data, portMAX_DELAY)) {
            snprintf(json, sizeof(json),
                "{\"temp\":%.2f,\"hum\":%.2f,\"press\":%.1f}",
                data.temperature / 100.0f,
                data.humidity    / 1024.0f,
                data.pressure    / 256.0f / 100.0f);
            esp8266_publish("env/node1", json);
        }
    }
}

int main(void)
{
    system_init();
    bme280_init();
    esp8266_init();

    xSensorQueue = xQueueCreate(4, sizeof(bme280_data_t));

    xTaskCreate(sensor_task, "Sensor", 512, NULL, 2, NULL);
    xTaskCreate(mqtt_task,   "MQTT",   768, NULL, 2, NULL);
    xTaskCreate(led_task,    "LED",    128, NULL, 1, NULL);

    vTaskStartScheduler();
    for (;;);
}
省電優化
啟用 FreeRTOS Tickless Idle 後，sensor_task 的 vTaskDelay(10000 ms) 會讓 CPU 進入 Stop Mode 長達 10 秒，平均電流可從 ~12 mA 降至 ~0.3 mA， 符合 REQ-03 的 5 mA 預算目標。
8.3

直流馬達 PID 控制器

編碼器測速、PWM 輸出、離散 PID 演算法與串列調參介面

閉迴路控制架構
PID 閉迴路控制流程 Setpoint (目標轉速) │ ▼ [+] ──────────────────────────────────────────┐ │ e(t) │ ▼ │ PID Controller → PWM Driver → DC Motor │ (Kp, Ki, Kd) (TIM1 CH1) (H-Bridge) │ │ 實際轉速 Encoder ←──────────────────┘ (TIM3 Encoder Mode)
編碼器測速（TIM3 Encoder Mode）

STM32 定時器的 Encoder Mode 可硬體計數 A/B 相脈波，不需要 CPU 中斷。 每 10 ms 讀取計數差值 ΔCount，換算轉速： RPM = (ΔCount / PPR / 4) × (60 / 0.01)， 其中 PPR 為編碼器每轉脈波數（本例使用 334 PPR）， ×4 為四倍頻計數。

C — encoder.c 
#define ENC_PPR    334     /* 每轉脈波數 */
#define SAMPLE_MS  10      /* 採樣週期 (ms) */

static int32_t last_count = 0;

void encoder_init(void)
{
    /* TIM3: Encoder Interface Mode，自動重載 0xFFFF */
    RCC->APB1ENR |= RCC_APB1ENR_TIM3EN;
    TIM3->SMCR  = TIM_SMCR_SMS_0 | TIM_SMCR_SMS_1;  /* Mode 3 */
    TIM3->CCMR1 = TIM_CCMR1_CC1S_0 | TIM_CCMR1_CC2S_0;
    TIM3->ARR   = 0xFFFF;
    TIM3->CR1  |= TIM_CR1_CEN;
}

float encoder_get_rpm(void)
{
    int32_t now   = (int16_t)TIM3->CNT;  /* 有符號讀取支援反轉 */
    int32_t delta = now - last_count;
    last_count    = now;

    /* 換算：ΔCount / (PPR×4) 轉 × (60000/SAMPLE_MS) ms/min */
    return ((float)delta / (ENC_PPR * 4)) * (60000.0f / SAMPLE_MS);
}
離散 PID 演算法

位置式 PID 公式： u(k) = Kp·e(k) + Ki·Σe·T + Kd·(e(k)-e(k-1))/T。 實作時需加入積分飽和限制（Anti-Windup）與輸出限幅， 防止突然給定大誤差時 PWM 超出 100%。

C — pid.c 
typedef struct {
    float kp, ki, kd;
    float integral;
    float prev_error;
    float integral_limit;  /* Anti-Windup 積分上限 */
    float output_limit;    /* 輸出飽和上限（0–100%） */
} pid_t;

float pid_compute(pid_t *pid, float setpoint, float measured, float dt)
{
    float error = setpoint - measured;

    /* P 項 */
    float p_term = pid->kp * error;

    /* I 項（含 Anti-Windup 限幅） */
    pid->integral += error * dt;
    if      (pid->integral >  pid->integral_limit) pid->integral =  pid->integral_limit;
    else if (pid->integral < -pid->integral_limit) pid->integral = -pid->integral_limit;
    float i_term = pid->ki * pid->integral;

    /* D 項（微分項，使用前後差） */
    float d_term = pid->kd * (error - pid->prev_error) / dt;
    pid->prev_error = error;

    float output = p_term + i_term + d_term;

    /* 輸出限幅 */
    if      (output >  pid->output_limit) output =  pid->output_limit;
    else if (output < -pid->output_limit) output = -pid->output_limit;

    return output;
}
串列調參介面（CLI over UART）

量產前的調參通常需要反覆修改 Kp/Ki/Kd 並重刷韌體，效率極低。 改用 UART CLI 介面，可在執行期動態設定參數並即時觀察響應曲線， 搭配 Python 腳本可自動繪製 step response 圖。

UART — 調參指令範例 
# 設定目標轉速 150 RPM
set sp 150

# 調整 PID 係數
set kp 0.8
set ki 0.15
set kd 0.05

# 查詢目前轉速與誤差
get status
# → sp=150.00 rpm=148.73 err=1.27 duty=62.4%

# 串流輸出（每 10 ms 一筆，CSV 格式給 Python 繪圖）
stream on
# → 0,150.00,0.00,0.0
# → 10,150.00,23.41,18.2
# → 20,150.00,89.12,55.6
# → ...
調參建議
先用 Ziegler-Nichols 臨界增益法：令 Ki=Kd=0，逐步增大 Kp 直到出現持續振盪， 記錄臨界增益 Ku 與振盪週期 Tu，再依表設定 Kp=0.6Ku、Ki=1.2Ku/Tu、Kd=0.075Ku·Tu。 此後再微調以消除超射或穩態誤差。

Python 資料橋接腳本

使用 paho-mqtt 訂閱感測資料，解析 JSON 後寫入 InfluxDB， 提供時序資料的長期儲存與查詢能力。

Python — mqtt_bridge.py 
import json
import paho.mqtt.client as mqtt
from influxdb_client import InfluxDBClient, Point
from influxdb_client.client.write_api import SYNCHRONOUS

BROKER   = "localhost"
TOPIC    = "env/#"
INFLUX_URL   = "http://localhost:8086"
INFLUX_TOKEN = "my-token"
INFLUX_ORG   = "lab"
INFLUX_BKT   = "sensors"

client_influx = InfluxDBClient(url=INFLUX_URL, token=INFLUX_TOKEN, org=INFLUX_ORG)
write_api = client_influx.write_api(write_options=SYNCHRONOUS)

def on_message(client, userdata, msg):
    try:
        data  = json.loads(msg.payload)
        node  = msg.topic.split("/")[-1]  # e.g. "node1"
        point = (Point("environment")
                 .tag("node", node)
                 .field("temperature", data["temp"])
                 .field("humidity",    data["hum"])
                 .field("pressure",    data.get("press", 0)))
        write_api.write(bucket=INFLUX_BKT, record=point)
        print(f"[{node}] T={data['temp']:.1f}°C H={data['hum']:.1f}%")
    except Exception as e:
        print(f"Error: {e}")

mqttc = mqtt.Client()
mqttc.on_message = on_message
mqttc.connect(BROKER, 1883)
mqttc.subscribe(TOPIC)
mqttc.loop_forever()

嵌入式系統的測試分三層： 單元測試在 Host PC 上跑（用 Unity/CppUTest 框架，不需硬體）； 整合測試在目標板上跑（驗證驅動與 HAL 接層）； 系統測試含完整硬體鏈路，以 Python 腳本自動化驗收。

PID 控制器單元測試
C — test_pid.c（Unity 框架） 
#include "unity.h"
#include "pid.h"

static pid_t g_pid;

void setUp(void)
{
    g_pid = (pid_t){ .kp = 1.0f, .ki = 0.1f, .kd = 0.05f,
                     .integral_limit = 50.0f, .output_limit = 100.0f };
}

void test_zero_error_gives_zero_output(void)
{
    float out = pid_compute(&g_pid, 100.0f, 100.0f, 0.01f);
    TEST_ASSERT_FLOAT_WITHIN(0.001f, 0.0f, out);
}

void test_output_clamped_at_limit(void)
{
    /* 大誤差應被限制在 100% 以內 */
    float out = pid_compute(&g_pid, 1000.0f, 0.0f, 0.01f);
    TEST_ASSERT_LESS_OR_EQUAL_FLOAT(100.0f, out);
}

void test_integral_windup_clamped(void)
{
    /* 反覆累積後 integral 不超過 limit */
    for (int i = 0; i < 1000; i++)
        pid_compute(&g_pid, 100.0f, 0.0f, 0.01f);
    TEST_ASSERT_LESS_OR_EQUAL_FLOAT(50.0f, g_pid.integral);
}

int main(void)
{
    UNITY_BEGIN();
    RUN_TEST(test_zero_error_gives_zero_output);
    RUN_TEST(test_output_clamped_at_limit);
    RUN_TEST(test_integral_windup_clamped);
    return UNITY_END();
}

import json, time, pytest
import paho.mqtt.client as mqtt

BROKER  = "192.168.1.100"
TIMEOUT = 30  # 秒，等待節點上傳

received = []

def on_message(client, userdata, msg):
    received.append(json.loads(msg.payload))

def test_sensor_data_received():
    """REQ-01 & REQ-02: 應在 30 秒內收到至少一筆感測資料"""
    mqttc = mqtt.Client()

        time.sleep(0.5)
    mqttc.loop_stop()

def test_sensor_data_schema():
    """資料必須包含 temp, hum, press 欄位且在合理範圍內"""

    assert "temp"  in msg
    assert "hum"   in msg
