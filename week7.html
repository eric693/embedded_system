7.1

STM32 低功耗模式

三種低功耗狀態的硬體機制、喚醒來源與電流比較

三種低功耗模式概覽

STM32F446RE 提供三層次低功耗模式，關閉的電路域逐層擴大： Sleep 只停止 CPU 時脈； Stop 關閉所有時脈但保留 SRAM 與暫存器內容； Standby 關閉幾乎所有電源域，僅 Backup Domain 存活。

電源模式層次 ┌──────────────────────────────────────────────────────┐ │ Run Mode CPU + 所有外設 + PLL 全速運行 │ │ 典型功耗：~45 mA @ 180 MHz, VDD=3.3V │ ├──────────────────────────────────────────────────────┤ │ Sleep Mode CPU 時脈停止，外設繼續運作 │ │ WFI/WFE 指令進入；任意中斷喚醒 │ │ 典型功耗：~10–25 mA（視啟用外設而定） │ ├──────────────────────────────────────────────────────┤ │ Stop Mode 所有時脈停止（含外設），Vcore 可降壓 │ │ 喚醒來源：EXTI 線、RTC Alarm/Tamper/WakeUp │ │ 喚醒後 HSI 16 MHz 先啟動，需重新設定 PLL │ │ 典型功耗：~100–400 µA │ ├──────────────────────────────────────────────────────┤ │ Standby Mode 僅 Backup Domain 供電 │ │ 喚醒來源：WKUP pin (PA0)、RTC Alarm、NRST │ │ SRAM 內容遺失，相當於硬體重置 │ │ 典型功耗：~2–10 µA │ └──────────────────────────────────────────────────────┘
進入 Stop Mode — 暫存器操作

進入 Stop Mode 需設定 PWR_CR 的 LPDS（Low-Power Deepsleep）位， 以及 SCB 的 SLEEPDEEP 位，再執行 WFI。 為避免電壓調節器持續全速供電，建議同時設定 PDDS=0（Stop，非 Standby）。

C — stop_mode.c 
#include "stm32f4xx.h"

/* 進入 Stop Mode，以 EXTI line 13 (PC13 user button) 喚醒 */
void enter_stop_mode(void)
{
    /* 1. 關閉不需要的外設時鐘以降低 Stop 電流 */
    RCC->AHB1ENR &= ~(RCC_AHB1ENR_GPIOBEN | RCC_AHB1ENR_GPIOCEN);

    /* 2. 設定 PWR：電壓調節器進 Low-Power 模式（LPDS=1），PDDS=0 → Stop */
    PWR->CR |= PWR_CR_LPDS;
    PWR->CR &= ~PWR_CR_PDDS;

    /* 3. SCB SLEEPDEEP = 1 → 讓 WFI 觸發 Deepsleep */
    SCB->SCR |= SCB_SCR_SLEEPDEEP_Msk;

    /* 4. 確保 EXTI line 13 已設定為中斷來源（見 lesson 2.3） */
    __DSB();  /* Data Sync Barrier：確保所有寫入完成 */
    __ISB();  /* Instruction Sync Barrier */
    __WFI();  /* 進入 Stop，等待中斷 */

    /* 5. 喚醒後，硬體自動以 HSI 16 MHz 執行，需重新設定 PLL → 180 MHz */
    SCB->SCR &= ~SCB_SCR_SLEEPDEEP_Msk;
    SystemClock_Config();  /* 重新初始化 PLL */
}

/* EXTI15_10 中斷服務（PC13 按鈕喚醒） */
void EXTI15_10_IRQHandler(void)
{
    if (EXTI->PR & EXTI_PR_PR13) {
        EXTI->PR = EXTI_PR_PR13;  /* 清除 pending flag */
    }
    /* 不需要額外動作：中斷本身喚醒 CPU，main 繼續執行 */
}
RTC 定時喚醒 (Standby)

若應用需要定期醒來採樣後再次入眠，Standby + RTC AlarmA 是最省電的方案。 備份暫存器（BKP registers）可在 Standby 後保留少量狀態（如採樣計數）。

C — rtc_wakeup.c 
/* RTC 每 5 秒喚醒一次，量測後進入 Standby */

void rtc_wakeup_config(uint32_t seconds)
{
    /* 啟用 PWR 時鐘，解鎖 Backup Domain */
    RCC->APB1ENR |= RCC_APB1ENR_PWREN;
    PWR->CR      |= PWR_CR_DBP;

    /* 使用 LSI ~32 kHz 作為 RTC 時鐘來源 */
    RCC->CSR     |= RCC_CSR_LSION;
    while (!(RCC->CSR & RCC_CSR_LSIRDY));
    RCC->BDCR     = (RCC->BDCR & ~RCC_BDCR_RTCSEL) | RCC_BDCR_RTCSEL_1;
    RCC->BDCR    |= RCC_BDCR_RTCEN;

    /* 解除 RTC 寫保護 */
    RTC->WPR = 0xCA;
    RTC->WPR = 0x53;

    /* 使用 Wakeup Timer：WUCKSEL=100b → ck_spre (1 Hz) */
    RTC->CR  &= ~RTC_CR_WUTE;
    while (!(RTC->ISR & RTC_ISR_WUTWF));
    RTC->WUTR = seconds - 1;
    RTC->CR   = (RTC->CR & ~RTC_CR_WUCKSEL) | (0x4 << RTC_CR_WUCKSEL_Pos);
    RTC->CR  |= RTC_CR_WUTIE | RTC_CR_WUTE;

    /* 鎖回 RTC 寫保護 */
    RTC->WPR = 0xFF;

    /* 設定 EXTI line 22（RTC Wakeup）為上升沿中斷 */
    EXTI->IMR  |= (1U << 22);
    EXTI->RTSR |= (1U << 22);
}

void enter_standby(void)
{
    /* 清除 Standby / Wakeup flags */
    PWR->CR  |= PWR_CR_CSBF | PWR_CR_CWUF;
    PWR->CR  |= PWR_CR_PDDS;  /* PDDS=1 → Standby（非 Stop） */
    SCB->SCR |= SCB_SCR_SLEEPDEEP_Msk;
    __DSB();
    __WFI();
    /* 喚醒後等同硬體 Reset，程式從頭執行 */
}
低功耗模式比較表
模式 CPU 時脈 外設時脈 SRAM 喚醒時間 典型電流 (3.3 V) 喚醒來源
Run 全速 全速 保留 — ~42 mA (180 MHz) —
Sleep 停止 運行 保留 < 1 µs ~10–25 mA 任意中斷
Stop 停止 停止 保留 ~50 µs (HSI) ~100–400 µA EXTI, RTC
Standby 停止 停止 遺失 ~300 µs (Reset) ~2–8 µA WKUP, RTC, NRST
Stop 模式後 PLL 失效
從 Stop 模式喚醒後，HSI 16 MHz 自動恢復，但 PLL 已關閉。 若程式邏輯依賴 UART 鮑率或 Timer 週期正確，必須在喚醒後立即呼叫 SystemClock_Config() 重新啟動 PLL，否則所有時脈相依外設行為異常。

低功耗程式設計技巧

動態電壓頻率調整、外設時鐘閘控、FreeRTOS Tickless Idle

動態頻率調整 (Run Mode 省電)

並非所有任務都需要 180 MHz。 在等待感測器回應期間將 PLL 降為 16 MHz HSI，完成計算前再拉回 180 MHz， 可在不切換低功耗模式的情況下節省 30–50% 電流。 STM32 透過修改 RCC_CFGR.SW 切換時鐘來源，且切換過程不中斷外設。

C — clk_scale.c 
/* 切換至 HSI 16 MHz（省電），稍後再切回 PLL 180 MHz */

void clock_switch_hsi(void)
{
    /* 等待 HSI ready */
    RCC->CR |= RCC_CR_HSION;
    while (!(RCC->CR & RCC_CR_HSIRDY));

    /* 先降低 Flash 等待週期，再切換時脈，避免 Prefetch 錯誤 */
    FLASH->ACR = (FLASH->ACR & ~FLASH_ACR_LATENCY) | FLASH_ACR_LATENCY_0WS;

    /* 切換系統時鐘來源：SW = 00 → HSI */
    RCC->CFGR = (RCC->CFGR & ~RCC_CFGR_SW) | RCC_CFGR_SW_HSI;
    while ((RCC->CFGR & RCC_CFGR_SWS) != RCC_CFGR_SWS_HSI);

    /* 更新 SystemCoreClock 供 HAL_Delay / SysTick 正確使用 */
    SystemCoreClock = 16000000UL;
    SysTick_Config(SystemCoreClock / 1000);
}

void clock_switch_pll_180mhz(void)
{
    /* 先設好 Flash 等待週期 5WS (for 180 MHz) */
    FLASH->ACR = (FLASH->ACR & ~FLASH_ACR_LATENCY) | FLASH_ACR_LATENCY_5WS;
    FLASH->ACR |= FLASH_ACR_PRFTEN | FLASH_ACR_ICEN | FLASH_ACR_DCEN;

    /* 切回 PLL */
    RCC->CFGR = (RCC->CFGR & ~RCC_CFGR_SW) | RCC_CFGR_SW_PLL;
    while ((RCC->CFGR & RCC_CFGR_SWS) != RCC_CFGR_SWS_PLL);

    SystemCoreClock = 180000000UL;
    SysTick_Config(SystemCoreClock / 1000);
}
外設時鐘閘控

RCC 的 AHB1ENR / APB1ENR / APB2ENR 每個位元控制一個外設的時鐘。 未使用的外設時鐘應在初始化時明確關閉，這是 STM32 省電的最基本手段， 可在 Run Mode 下減少 2–5 mA（視外設數量）。

C — periph_clock_gate.c 
/* 僅啟用本應用需要的外設；其餘保持 reset 預設（多數為關閉） */

void periph_clock_init(void)
{
    /* AHB1：只開啟 GPIOA、GPIOB、DMA2 */
    RCC->AHB1ENR = RCC_AHB1ENR_GPIOAEN
                 | RCC_AHB1ENR_GPIOBEN
                 | RCC_AHB1ENR_DMA2EN;

    /* APB1：只開啟 USART2、I2C1 */
    RCC->APB1ENR = RCC_APB1ENR_USART2EN
                 | RCC_APB1ENR_I2C1EN;

    /* APB2：只開啟 ADC1、SPI1 */
    RCC->APB2ENR = RCC_APB2ENR_ADC1EN
                 | RCC_APB2ENR_SPI1EN;
}

/* 進入 Stop 前，關閉所有外設時鐘（除 RTC 外）以達最低 Stop 電流 */
void periph_clock_all_off(void)
{
    RCC->AHB1ENR = 0;
    RCC->APB1ENR = RCC_APB1ENR_PWREN;  /* PWR 必須保留 */
    RCC->APB2ENR = 0;
}
FreeRTOS Tickless Idle

FreeRTOS 預設每個 Tick（通常 1 ms）都會喚醒 CPU 更新計時器， 這在低功耗場景是很大的浪費。 啟用 configUSE_TICKLESS_IDLE = 1 後， Kernel 在所有任務都 Block 且下一個喚醒時間已知時， 會呼叫 vPortSuppressTicksAndSleep() 讓 MCU 進入 Sleep/Stop， 並在喚醒後以 RTC 或 SysTick 補回跳過的 Tick 數。

C — tickless_idle.c（portmacro 擴充） 
/* FreeRTOSConfig.h */
#define configUSE_TICKLESS_IDLE          2   /* 2 = 自訂實作 */
#define configEXPECTED_IDLE_TIME_BEFORE_SLEEP  5   /* 至少 5 tick 才值得入睡 */

/* port_tickless.c — 自訂 vPortSuppressTicksAndSleep */
void vPortSuppressTicksAndSleep(TickType_t xExpectedIdleTime)
{
    uint32_t ulReloadValue, ulCompleteTickPeriods;
    TickType_t xModifiableIdleTime;

    /* 計算 SysTick 重載值：讓 SysTick 在 idle 期間充當喚醒鬧鐘 */
    ulReloadValue = (configCPU_CLOCK_HZ / configTICK_RATE_HZ)
                    * xExpectedIdleTime - 1;

    portDISABLE_INTERRUPTS();

    /* 再次確認仍然 idle（中斷可能改變狀態） */
    xModifiableIdleTime = xExpectedIdleTime;
    configPRE_SLEEP_PROCESSING(xModifiableIdleTime);

    if (xModifiableIdleTime > 0) {
        /* 重新設定 SysTick 為長計時，然後進 Sleep */
        SysTick->LOAD = ulReloadValue;
        SysTick->VAL  = 0;
        __DSB();
        __WFI();  /* CPU 進入 Sleep，等待 SysTick 或其他中斷 */
    }

    configPOST_SLEEP_PROCESSING(xModifiableIdleTime);

    /* 計算實際睡了多少 tick，補回 xTickCount */
    ulCompleteTickPeriods = (ulReloadValue - SysTick->VAL)
                            / (configCPU_CLOCK_HZ / configTICK_RATE_HZ);
    vTaskStepTick(ulCompleteTickPeriods);

    /* 恢復 SysTick 為正常 1 ms 週期 */
    SysTick->LOAD = (configCPU_CLOCK_HZ / configTICK_RATE_HZ) - 1;
    SysTick->VAL  = 0;
    portENABLE_INTERRUPTS();
}
省電設計原則
「最省電的外設是沒有外設」。在設計硬體前先評估每個外設的靜態電流， 並為每個外設配備電源開關 FET（PMOS 或 load switch）， 讓韌體可以在不使用時完全斷電（例如 GPS、WiFi 模組的 EN 腳）。 這比任何軟體省電技巧都有效。
7.3

電源量測與分析

分流電阻電路、邏輯分析儀觸發、功耗時序圖解讀

硬體量測電路

最簡單的電流量測方法是在 VDD 供電線上串聯一個低阻值精密電阻（分流電阻，shunt）， 以差動放大器或示波器量測電壓差，換算電流。 Nucleo-F446RE 在 JP6（IDD 跳線）預留了此量測點：移除跳線，接入 0.1 Ω / 1 W 電阻即可。

量測電路（概念圖） 3.3V Rail MCU VDD ┤──── 0.1 Ω shunt ────────────── ┤ │ │ V+ V− └─── 差動放大器 ───→ 示波器 CH1 (Gain × 10) 量測計算： I = V_shunt / R_shunt = (V+ - V−) / 0.1 Ω 示波器讀 200 mV → I = 200 mV / 0.1 Ω = 2 A （舉例） 示波器讀 5 mV → I = 5 mV / 0.1 Ω = 50 mA GPIO 觸發方式（與邏輯分析儀配合）： 進入 Stop 前：PA8 HIGH → 邏輯分析儀 CH2 上升沿 Trigger 喚醒完成後： PA8 LOW → 標記 Stop 模式結束 可在電流波形上對齊找出實際 Stop 電流與喚醒尖峰
韌體埋點 (Power Markers)

透過 GPIO 輸出狀態讓示波器或邏輯分析儀區分各個工作階段， 這是「免費的」功耗 profiling 工具，不需要任何付費軟體。

C — power_marker.c 
/* PA8 作為 Power Marker，高電位 = 主動工作，低電位 = 省電 */
#define MARKER_HIGH()  (GPIOA->BSRR = GPIO_BSRR_BS8)
#define MARKER_LOW()   (GPIOA->BSRR = GPIO_BSRR_BR8)

int main(void)
{
    system_init();

    while (1) {
        MARKER_HIGH();       /* 示波器標記：開始工作 */
        sensor_read();       /* SPI 讀取感測器 ~2 ms */
        process_data();      /* DSP 計算 ~0.5 ms */
        uart_transmit();     /* 上傳結果 ~1 ms */
        MARKER_LOW();        /* 示波器標記：開始省電 */

        enter_stop_mode();   /* Stop ~10 s，由 RTC 喚醒 */
    }
}

量測陷阱
量測 Stop / Standby 電流時，若 Nucleo 板上的 ST-LINK 仍透過 USB 供電， ST-LINK 的 3.3 V LDO 會持續向 MCU 供電，使量測值偏高 2–10 mA。 正確做法是移除 JP5（3V3 jumper），由外部量測電路供電，ST-LINK 僅用於 SWD 觀察。

安全啟動與韌體保護

RDP 保護等級、Option Bytes、Chain of Trust、韌體完整性驗證

STM32 Read-Out Protection (RDP)

RDP 是 STM32 防止 Flash 被讀取的第一道防線。 透過 Option Bytes 的 RDP[7:0] 欄位設定：

等級 RDP 值 Flash 讀取 SRAM 讀取 (debug) 降級方式
Level 0 0xAA 允許 允許 出廠預設
Level 1 其他值 禁止（debugger 讀取） 允許（debug 仍可連） 降回 Level 0 → Mass Erase Flash
Level 2 0xCC 完全禁止 完全禁止 不可逆（晶片永久鎖定）
不可逆警告
Level 2 是永久性設定，無法解除，亦無法再 Flash 韌體更新。 量產前必須確認韌體完全穩定，並建立 OTA 更新機制。切勿在開發階段設定 Level 2。

Chain of Trust — 安全啟動流程

完整的安全啟動（Secure Boot）確保每個啟動階段都驗證下一階段的完整性， 任何竄改都會停止啟動。以 SHA-256 + HMAC 為例的流程如下：

安全啟動鏈（Secure Boot Chain） ROM Bootloader (不可修改，廠商燒錄) │ 驗證：Secure Boot Key hash 是否與 OTP 一致 ▼ Stage-1 Bootloader (Sector 0–3，WRP 保護) │ 功能：驗證 Application 的 SHA-256 摘要 │ 摘要儲存於：Flash 最後 32 bytes 或 Backup Register │ 金鑰來源：OTP (One-Time Programmable) 區域 ▼ Application Firmware (Sector 4–11) │ 正常執行 ▼ （OTA 更新時，Stage-1 驗證新韌體簽章後才覆寫）
韌體完整性驗證（SHA-256）
C — fw_verify.c（Bootloader 片段） 
#include "stm32f4xx.h"
#include "sha256.h"   /* 輕量 SHA-256 實作，或使用 STM32 HASH 硬體 */

#define APP_START_ADDR   0x08020000UL  /* Application 起始地址（Sector 4） */
#define APP_SIZE         0x00060000UL  /* 384 KB */
#define APP_HASH_ADDR    0x08080000UL  /* 儲存預期 SHA-256 的地址 */

typedef void (*app_entry_t)(void);

bool bootloader_verify_and_jump(void)
{
    uint8_t  computed[32];
    uint8_t *expected = (uint8_t *)APP_HASH_ADDR;

    /* 計算 Application 區域的 SHA-256 */
    sha256_compute((const uint8_t *)APP_START_ADDR, APP_SIZE, computed);

    /* 恆定時間比較（防止 timing side-channel attack） */
    uint8_t diff = 0;
    for (int i = 0; i < 32; i++) {
        diff |= computed[i] ^ expected[i];
    }
    if (diff != 0) {
        /* 驗證失敗：停機或進入 Recovery 模式 */
        while (1) { /* Halt — 不跳轉 */ }
    }

    /* 驗證通過：設定 MSP 並跳轉到 Application */
    uint32_t   app_sp    = *((uint32_t *)APP_START_ADDR);
    uint32_t   app_reset = *((uint32_t *)(APP_START_ADDR + 4));
    app_entry_t jump     = (app_entry_t)app_reset;

    __set_MSP(app_sp);
    SCB->VTOR = APP_START_ADDR;  /* 重定向 Vector Table */
    jump();

    return true;  /* unreachable */
}

硬體加密加速器

STM32 AES、HASH、RNG 外設的暫存器操作與效能比較

硬體加速器概覽

STM32F437/F439 系列內建加密外設；F446 系列則無 CRYP 硬體， 但配備 HASH 與 RNG。 若需要 AES 加速，可選擇 STM32L4 Plus / STM32H7 系列， 或在 F4 上使用軟體實作（MbedTLS / TinyCrypt）。 以下以 STM32F437VG（含 CRYP）為示範目標。

加速器 支援演算法 DMA 支援 典型速度（CPU @ 180 MHz）
CRYP AES-128/192/256 (ECB/CBC/CTR), 3DES, DES 是 >500 Mbps（vs 軟體 ~50 Mbps）
HASH MD5, SHA-1, SHA-224, SHA-256, HMAC 是 >700 Mbps
RNG TRNG（熱雜訊源） 否 ~1 Mbit/s
AES-CBC 加密（CRYP 外設）
C — aes_cbc.c 
#include "stm32f4xx.h"

/* AES-128-CBC 加密（CRYP 硬體外設） */
void aes_cbc_encrypt(const uint32_t *key,    /* 128-bit, 4 words */
                     const uint32_t *iv,     /* 128-bit IV        */
                     const uint32_t *plain,  /* 明文（16 bytes 倍數）*/
                     uint32_t       *cipher, /* 密文輸出          */
                     uint32_t        blocks) /* 資料塊數（每塊 16B）*/
{
    /* 啟用 CRYP 時鐘 */
    RCC->AHB2ENR |= RCC_AHB2ENR_CRYPEN;

    /* 重置 CRYP */
    CRYP->CR = CRYP_CR_FFLUSH;

    /* 設定：AES-128-CBC 加密模式，32-bit data 對齊 */
    CRYP->CR = (CRYP_CR_ALGOMODE_AES_CBC)   /* 0b111 << 3 */
             | (CRYP_CR_DATATYPE_32B)        /* 0b00 << 6：不交換 byte */
             | (CRYP_CR_KEYSIZE_128B);       /* 0b00 << 8 */

    /* 載入金鑰（K0R ~ K3R，從高位開始）*/
    CRYP->K0LR = key[0]; CRYP->K0RR = key[1];
    CRYP->K1LR = key[2]; CRYP->K1RR = key[3];

    /* 載入 IV（初始向量）*/
    CRYP->IV0LR = iv[0]; CRYP->IV0RR = iv[1];
    CRYP->IV1LR = iv[2]; CRYP->IV1RR = iv[3];

    /* 啟動 CRYP */
    CRYP->CR |= CRYP_CR_CRYPEN;

    for (uint32_t b = 0; b < blocks; b++) {
        const uint32_t *in  = plain  + b * 4;
        uint32_t       *out = cipher + b * 4;

        /* 等待輸入 FIFO 有空間 */
        while (!(CRYP->SR & CRYP_SR_IFNF));
        CRYP->DIN = in[0]; CRYP->DIN = in[1];
        CRYP->DIN = in[2]; CRYP->DIN = in[3];

        /* 等待輸出 FIFO 有資料 */
        while (!(CRYP->SR & CRYP_SR_OFNE));
        out[0] = CRYP->DOUT; out[1] = CRYP->DOUT;
        out[2] = CRYP->DOUT; out[3] = CRYP->DOUT;
    }

    CRYP->CR &= ~CRYP_CR_CRYPEN;  /* 停止 CRYP */
}
真隨機數產生器 (RNG / TRNG)

RNG 外設以類比雜訊源（熱雜訊）產生真亂數， 通過 FIPS 140-2 統計測試。每次讀取前必須確認 DRDY 旗標， 並定期檢查 SEIS（Seed Error）及 CEIS（Clock Error）旗標確保亂數品質。

C — rng.c 
void rng_init(void)
{
    RCC->AHB2ENR |= RCC_AHB2ENR_RNGEN;
    RNG->CR      |= RNG_CR_RNGEN;
}

/* 取得一個 32-bit 真亂數，失敗回傳 false */
bool rng_get(uint32_t *out)
{
    uint32_t timeout = 1000;

    while (!(RNG->SR & RNG_SR_DRDY)) {
        if (--timeout == 0) return false;
    }

    /* 檢查 Seed / Clock 錯誤 */
    if (RNG->SR & (RNG_SR_SEIS | RNG_SR_CEIS)) {
        RNG->SR &= ~(RNG_SR_SEIS | RNG_SR_CEIS);
        return false;  /* 清除後重試 */
    }

    *out = RNG->DR;
    return true;
}

/* 產生 N bytes 的隨機資料（用於 IV、Nonce、Salt） */
void rng_fill(uint8_t *buf, size_t len)
{
    size_t   i = 0;
    uint32_t word;

    while (i < len) {
        if (!rng_get(&word)) continue;
        size_t copy = (len - i < 4) ? (len - i) : 4;
        __builtin_memcpy(buf + i, &word, copy);
        i += copy;
    }
}
SHA-256 硬體 HASH（F4 通用）
C — hash_sha256.c 
/* 使用 STM32 HASH 外設計算 SHA-256 摘要 */
void hash_sha256(const uint8_t *data, size_t len, uint8_t digest[32])
{
    RCC->AHB2ENR |= RCC_AHB2ENR_HASHEN;

    /* 設定：SHA-256，big-endian 字元組輸入 */
    HASH->CR = HASH_CR_ALGO_1    /* SHA-256 = bit[18:17] = 01 */
             | HASH_CR_DATATYPE_8B   /* 0b10：byte 反轉 */
             | HASH_CR_INIT;    /* 初始化 Hash 處理器 */

    /* 逐 word 寫入資料（HASH->DIN 每次接受 32 bits） */
    const uint32_t *p   = (const uint32_t *)data;
    size_t          words = len / 4;
    for (size_t i = 0; i < words; i++) {
        HASH->DIN = p[i];
    }

    /* 處理最後不足 4 bytes 的尾巴 */
    uint32_t tail = 0;
    size_t   rem  = len % 4;
    if (rem) {
        __builtin_memcpy(&tail, data + words * 4, rem);
        HASH->CR |= ((rem * 8) & HASH_CR_NBLW_Msk) << HASH_CR_NBLW_Pos;
        HASH->DIN = tail;
    }

    /* 觸發 padding 並等待完成 */
    HASH->STR = HASH_STR_DCAL;
    while (!(HASH->SR & HASH_SR_DCIS));

    /* 讀取 8 個 32-bit digest（SHA-256 = 256 bit = 32 bytes） */
    uint32_t *out = (uint32_t *)digest;
    out[0] = __REV(HASH->HR[0]);  /* __REV：byte-swap for big-endian */
    out[1] = __REV(HASH->HR[1]);
    out[2] = __REV(HASH->HR[2]);
    out[3] = __REV(HASH->HR[3]);
    out[4] = __REV(HASH_DIGEST->HR[0]);
    out[5] = __REV(HASH_DIGEST->HR[1]);
    out[6] = __REV(HASH_DIGEST->HR[2]);
    out[7] = __REV(HASH_DIGEST->HR[3]);
}

MPU — 記憶體保護單元

Cortex-M4 的 MPU 提供最多 8 個可配置的記憶體保護區域。 典型用途：將 Bootloader 區域設為唯讀、讓任務堆疊下方加一個「Guard Region」 以在堆疊溢位時觸發 MemManage Fault。

C — mpu_config.c 
/* 設定 MPU Region 0：Flash（唯讀執行）Region 1：Stack Guard */

void mpu_init(void)
{
    /* 關閉 MPU 以進行配置 */
    ARM_MPU_Disable();

    /* Region 0：整個 Flash（0x0800_0000，2 MB），唯讀可執行 */
    MPU->RNR  = 0;
    MPU->RBAR = 0x08000000;
    MPU->RASR = ARM_MPU_RASR(
        0,                           /* XN=0: 允許執行 */
        ARM_MPU_AP_RO,               /* AP: Read-Only */
        ARM_MPU_TEX_NORMAL_WT,       /* Write-Through Cache */
        1, 0,                        /* S=1(Shareable), C=0 */
        0,                           /* Sub-region disable = 0 */
        ARM_MPU_REGION_SIZE_2MB,     /* 2 MB */
        1                            /* Enable */
    );

    /* Region 1：Stack Guard（32 bytes，位於 Task Stack 底部）
     * 設為 No Access：若 Stack 溢出碰到此區域 → MemManage Fault */
    extern uint32_t _task_stack_bottom;  /* linker symbol */
    MPU->RNR  = 1;
    MPU->RBAR = (uint32_t)&_task_stack_bottom;
    MPU->RASR = ARM_MPU_RASR(
        1,                           /* XN=1: 禁止執行 */
        ARM_MPU_AP_NONE,             /* AP: No Access */
        ARM_MPU_TEX_NORMAL_WT,
        1, 0,
        0,
        ARM_MPU_REGION_SIZE_32B,     /* 32 bytes Guard */
        1
    );

    /* 啟動 MPU，啟用 PRIVDEFENA（特權模式保留預設 map） */
    ARM_MPU_Enable(MPU_CTRL_PRIVDEFENA_Msk);
}

/* MemManage Fault Handler：輸出故障位址並停機 */
void MemManage_Handler(void)
{
    uint32_t mmfar = SCB->MMFAR;  /* 發生保護違規的地址 */
    uint8_t  mmfsr = SCB->CFSR & 0xFF;

    /* 記錄到 Backup RAM 或 UART，然後系統重置 */
    fault_log(mmfar, mmfsr);
    NVIC_SystemReset();
}
Stack Canary 與安全編譯旗標

GCC 的 -fstack-protector-strong 在函數 prologue 寫入隨機 Canary 值，epilogue 驗證，若被覆寫則呼叫 __stack_chk_fail()。嵌入式環境需自行實作此函數。

Makefile — 安全編譯旗標 
CFLAGS += \
    -fstack-protector-strong \     # Stack Canary（推薦）
    -fno-stack-protector \         # 在 ISR / startup 用 __attribute__((no_stack_protector))
    -D_FORTIFY_SOURCE=2 \          # buffer overflow 偵測（需 libc 支援）
    -Wformat=2 \                   # printf format-string 漏洞警告
    -Warray-bounds \               # 陣列越界存取警告
    -fno-common \                  # 禁止弱符號 BSS 合併（降低 symbol collision 風險）
    -fPIE                          # Position Independent Executable（嵌入式通常不用，Linux 用）

LDFLAGS += \
    -Wl,-z,noexecstack \           # 禁止 stack 執行
    -Wl,-z,relro \                 # 唯讀重定位段（ELF）
    -Wl,-z,now                     # 立即解析所有符號
C — stack_check_fail.c（嵌入式實作） 
/* GCC 要求提供此符號；預設 libc 版本呼叫 abort()，嵌入式需自訂 */
#include "stm32f4xx.h"
#include <stdint.h>

void __stack_chk_fail(void)
{
    /* 記錄事件到 Backup Register（Standby 後仍可讀） */
    RCC->APB1ENR |= RCC_APB1ENR_PWREN;
    PWR->CR      |= PWR_CR_DBP;
    RTC->BKP0R    = 0xDEADC0DE;  /* 事件標記 */

    NVIC_SystemReset();  /* 立即系統重置 */
}

/* 必須提供初始 Canary 值（通常由啟動程式碼以 RNG 初始化） */
uintptr_t __stack_chk_guard;

void stack_guard_init(void)
{
    uint32_t rng_val;
    rng_init();
    rng_get(&rng_val);
    __stack_chk_guard = (uintptr_t)rng_val;
}

Buffer Overflow strcpy, memcpy 無邊界檢查 使用 strncpy / strlcpy；-fstack-protector-strong；MPU Guard Region
