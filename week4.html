4.1 — 理論 + 實作
UART / USART
非同步串列通訊、鮑率計算、DMA 環形緩衝、流控
UART 訊框格式與電氣特性

UART（Universal Asynchronous Receiver/Transmitter）以非同步方式傳輸， 無共用時脈，雙方必須預先協議相同的鮑率（Baud Rate）。 每個訊框由 1 個起始位元（低電位）、5–9 個資料位元、可選的同位元以及 1–2 個停止位元（高電位）組成。

Idle Start D0 D1 D2 D3 D4 D5 D6 D7 Parity Stop Idle 1 │ 0 │ x x x x x x x x │ x │ 1 │ 1 ▲ ▲ 下降緣觸發採樣開始 1 or 2 bit
鮑率暫存器計算（BRR）

STM32 的 USART 鮑率由 BRR 暫存器決定，計算公式如下：

BRR 計算公式
Over8 = 0（預設，16 倍過取樣）：
BRR = f_PCLK / Baud

範例：USART2 掛在 APB1（45 MHz），目標 115200 baud：
BRR = 45,000,000 / 115200 ≈ 390.6 → 0x0186 + 小數部分

實務上直接讓 HAL 計算，或使用 STM32CubeMX 確認誤差 < 2%。
以 DMA 實作環形緩衝接收
c — USART2 + DMA1 Stream5 環形緩衝接收 
#define RX_BUF_SIZE  256

static uint8_t  rx_buf[RX_BUF_SIZE];
static uint16_t rx_tail = 0;   /* 上次處理到的位置 */

void uart2_dma_init(void)
{
    /* 1. GPIO：PA2=TX(AF7), PA3=RX(AF7) — 省略，見 Week3 */

    /* 2. 啟用 USART2 時脈 */
    RCC->APB1ENR |= RCC_APB1ENR_USART2EN;

    /* 3. 設定 USART2：115200 8N1 */
    USART2->BRR = 0x0186;          /* 45MHz / 115200 ≈ 390 */
    USART2->CR1 = USART_CR1_UE    /* 啟用 USART */
                | USART_CR1_RE    /* 接收器 enable */
                | USART_CR1_TE;   /* 傳送器 enable */
    USART2->CR3 |= USART_CR3_DMAR; /* 啟用 RX DMA 請求 */

    /* 4. DMA1 Stream5 Channel4 = USART2_RX */
    RCC->AHB1ENR |= RCC_AHB1ENR_DMA1EN;
    DMA1_Stream5->CR = 0;
    while (DMA1_Stream5->CR & DMA_SxCR_EN);

    DMA1_Stream5->PAR  = (uint32_t)&USART2->DR;
    DMA1_Stream5->M0AR = (uint32_t)rx_buf;
    DMA1_Stream5->NDTR = RX_BUF_SIZE;
    DMA1_Stream5->CR   =
        (4U << DMA_SxCR_CHSEL_Pos) |  /* Channel 4 */
        DMA_SxCR_MINC               |  /* Memory 遞增 */
        DMA_SxCR_CIRC;                 /* Circular（環形）*/
    DMA1_Stream5->CR |= DMA_SxCR_EN;
}

/* 主迴圈輪詢：計算 DMA 已寫入多少新資料 */
void uart2_process(void)
{
    uint16_t head = RX_BUF_SIZE - DMA1_Stream5->NDTR; /* DMA 目前寫到哪 */
    while (rx_tail != head) {
        handle_byte(rx_buf[rx_tail]);
        rx_tail = (rx_tail + 1) % RX_BUF_SIZE;
    }
}
Idle Line 中斷搭配 DMA
啟用 USART_CR1_IDLEIE 後，匯流排空閒時會觸發中斷， 此時可立即處理緩衝區中的不定長度封包，不需等 DMA 滿才處理。 這是嵌入式 UART 協定解析的最佳實踐（STM32 HAL 的 UART_RECEIVER_TIMEOUT 亦基於此）。
4.2 — 理論 + 實作
SPI
全雙工同步串列、四種模式（CPOL/CPHA）、NSS 管理、DMA 加速
SPI 訊號線與時序模式

SPI 使用四條訊號線：SCK（時脈）、MOSI（Master Out Slave In）、 MISO（Master In Slave Out）、CS/NSS（片選，低有效）。 時序模式由 CPOL（時脈極性）與 CPHA（相位）決定，共四種組合。

Mode 101低下降緣SSD1306（部分）
Mode 210高下降緣—
Mode 311高上升緣SD Card、MPU-6000
驅動 W25Q128 SPI Flash（讀取 JEDEC ID）
c — SPI1 Mode 0，讀取 W25Q128 裝置 ID 
/* SPI1: PA5=SCK, PA6=MISO, PA7=MOSI（AF5），PA4=CS（GPIO 輸出）*/

static inline void cs_low(void)  { GPIOA->BSRR = (1U << (4+16)); }
static inline void cs_high(void) { GPIOA->BSRR = (1U << 4); }

uint8_t spi_transfer(uint8_t data)
{
    while (!(SPI1->SR & SPI_SR_TXE));   /* 等待 TX 緩衝空 */
    *(volatile uint8_t *)&SPI1->DR = data;
    while (!(SPI1->SR & SPI_SR_RXNE));  /* 等待 RX 有資料 */
    return *(volatile uint8_t *)&SPI1->DR;
}

void w25q_read_jedec_id(uint8_t *mfr, uint8_t *type, uint8_t *cap)
{
    cs_low();
    spi_transfer(0x9F);   /* JEDEC ID 指令 */
    *mfr  = spi_transfer(0xFF);  /* Manufacturer（W25Q = 0xEF）*/
    *type = spi_transfer(0xFF);  /* Memory Type */
    *cap  = spi_transfer(0xFF);  /* Capacity（0x18 = 128Mbit）*/
    cs_high();
}

void spi1_init(void)
{
    RCC->APB2ENR |= RCC_APB2ENR_SPI1EN;
    /* MSTR=1, BR=101(÷64 → ~1.4MHz), SSM=1, SSI=1, SPE=1 */
    SPI1->CR1 = SPI_CR1_MSTR | (5U << SPI_CR1_BR_Pos) |
                SPI_CR1_SSM  | SPI_CR1_SSI | SPI_CR1_SPE;
    SPI1->CR2 = 0;   /* 8-bit 資料幀 */
}
8-bit vs. 16-bit 資料幀存取
STM32F4 的 SPI DR 暫存器是 16-bit，但 FIFO 由存取寬度決定幀大小。 8-bit 模式時必須用 (volatile uint8_t *)&SPI1->DR 存取， 否則會送出 16-bit 幀。STM32F7/H7 有更明確的 FRXTH 位元控制。
4.3 — 理論 + 實作
I²C
半雙工兩線協定、位址、ACK/NACK、Clock Stretching、多主機
I²C 協定要點

I²C 使用兩條開汲（Open-Drain）訊號線：SDA（資料）與 SCL（時脈）， 需要外部上拉電阻（通常 4.7 kΩ）。所有裝置並聯在匯流排上，靠 7-bit 或 10-bit 從機位址區分目標。標準速度 100 kHz，快速模式 400 kHz，快速模式+ 1 MHz。

典型 I²C 寫入時序（Master 寫 1 byte 到 Slave） SDA: Start │ Addr[6:0] │ W │ ACK │ Data[7:0] │ ACK │ Stop SCL: ▔▔ │ ▔▔▔▔▔▔▔ │ │ ▁▁ │ ▔▔▔▔▔▔▔ │ ▁▁ │ ▔▔ Start = SDA 下降緣（SCL 高時） Stop = SDA 上升緣（SCL 高時） ACK = 從機將 SDA 拉低（第 9 個時脈） NACK = SDA 保持高（裝置不存在或忙碌）
讀寫 BME280 溫溼壓感測器
c — I²C1 讀取 BME280 Chip ID（暫存器 0xD0） 
#define BME280_ADDR  (0x76 << 1)  /* 7-bit 位址左移 1 bit */
#define BME280_ID_REG  0xD0

/* 等待旗標並逾時保護 */
static int i2c_wait(uint32_t flag, uint32_t set)
{
    uint32_t t = 10000;
    if (set) { while (!(I2C1->SR1 & flag) && --t); }
    else     { while  (I2C1->SR1 & flag  && --t); }
    return t ? 0 : -1;   /* 0 = OK, -1 = timeout */
}

int i2c_read_reg(uint8_t dev_addr, uint8_t reg, uint8_t *data)
{
    /* Phase 1: 寫入暫存器位址 */
    I2C1->CR1 |= I2C_CR1_START;
    if (i2c_wait(I2C_SR1_SB, 1)) return -1;      /* 等待 Start */

    I2C1->DR = dev_addr;                            /* 送位址（寫模式）*/
    if (i2c_wait(I2C_SR1_ADDR, 1)) return -1;    /* 等待 ADDR */
    (void)I2C1->SR2;                                /* 讀 SR2 清除 ADDR */

    I2C1->DR = reg;
    if (i2c_wait(I2C_SR1_TXE, 1)) return -1;

    /* Phase 2: Repeated Start → 讀取 1 byte */
    I2C1->CR1 |= I2C_CR1_START;
    if (i2c_wait(I2C_SR1_SB, 1)) return -1;

    I2C1->CR1 &= ~I2C_CR1_ACK;                     /* 最後 1 byte 不回 ACK */
    I2C1->DR = dev_addr | 0x01;                   /* 讀模式 */
    if (i2c_wait(I2C_SR1_ADDR, 1)) return -1;
    (void)I2C1->SR2;
    I2C1->CR1 |= I2C_CR1_STOP;                     /* 提前設定 Stop */

    if (i2c_wait(I2C_SR1_RXNE, 1)) return -1;
    *data = (uint8_t)I2C1->DR;
    I2C1->CR1 |= I2C_CR1_ACK;                     /* 恢復 ACK 給下次傳輸 */
    return 0;
}
I²C 匯流排鎖死（Bus Lockup）
若通訊中途 Reset，Slave 可能仍在等待更多時脈而將 SDA 拉低， 導致 Master 無法送出 Start 條件（SDA 被占用）。 解法：在 GPIO 模式下手動產生 9 個 SCL 脈衝，強制 Slave 釋放 SDA， 再重新初始化 I²C 周邊。
4.4 — 理論 + 實作
Timer 與 PWM
通用計時器架構、PWM 解析度、互補輸出與 DMA Burst（WS2812）
通用計時器架構

STM32F446 的 TIM2–TIM5 為 32-bit 通用計時器，掛在 APB1 上（Timer clock = 90 MHz）。 核心由 預分頻器（PSC）、自動重載暫存器（ARR） 與 計數器（CNT） 組成。

PWM 關鍵參數計算
PWM 頻率：f_PWM = f_TIM / ((PSC+1) × (ARR+1))
解析度：ARR + 1 個計數步階
工作週期：Duty = CCR / (ARR+1)

範例：產生 50 Hz 伺服馬達 PWM（TIM2，90 MHz）：
PSC = 89（÷90，1 MHz tick）
ARR = 19999（20 ms 週期，1–2 ms pulse = CCR 1000–2000）
配置 TIM3 CH1 輸出 PWM（PA6）
c — TIM3 CH1 PWM，50 Hz 伺服控制 
void tim3_pwm_init(void)
{
    /* PA6 = TIM3_CH1（AF2）*/
    RCC->AHB1ENR  |= RCC_AHB1ENR_GPIOAEN;
    RCC->APB1ENR  |= RCC_APB1ENR_TIM3EN;

    GPIOA->MODER  &= ~(3U << (6*2));
    GPIOA->MODER  |=  (2U << (6*2));  /* AF 模式 */
    GPIOA->AFR[0] |=  (2U << (6*4));  /* AF2 = TIM3 */

    TIM3->PSC   = 89;        /* 90MHz ÷ 90 = 1 MHz */
    TIM3->ARR   = 19999;     /* 50 Hz */
    TIM3->CCR1  = 1500;      /* 1.5 ms = 中間位置 */

    /* PWM Mode 1：CNT < CCR → 高電位 */
    TIM3->CCMR1 = (6U << TIM_CCMR1_OC1M_Pos) | TIM_CCMR1_OC1PE;
    TIM3->CCER  |= TIM_CCER_CC1E;  /* 啟用 CH1 輸出 */
    TIM3->CR1   |= TIM_CR1_ARPE | TIM_CR1_CEN;
    TIM3->EGR   |= TIM_EGR_UG;     /* 觸發更新，載入 PSC/ARR 預載值 */
}

/* 設定工作週期（pulse_us 單位：微秒，1000–2000 對應 0°–180°）*/
void servo_set_us(uint16_t pulse_us) { TIM3->CCR1 = pulse_us; }
DMA Burst 驅動 WS2812B LED 燈條

WS2812B 採用單線 800 kHz NRZ 協定：邏輯 "0" = 短高電位 + 長低電位， 邏輯 "1" = 長高電位 + 短低電位。利用 TIM 的 PWM + DMA Burst， 以 ARR 固定週期 1.25 µs，每個 bit 對應一個 CCR 值， 即可不佔用 CPU 驅動任意數量的 LED。

WS2812 DMA 技術要點
• ARR = 89（90 MHz ÷ 90 = 1 MHz，周期 1 µs → 搭配 800 kHz 需調整）
• 實際 800 kHz：ARR = 112（90 MHz ÷ 113 ≈ 796 kHz ≈ 1.25 µs）
• "0" bit：CCR ≈ 28（T0H ≈ 350 ns）
• "1" bit：CCR ≈ 58（T1H ≈ 700 ns）
• DMA 在每個更新事件（Update Event）時自動更新下一個 CCR， 實現 CPU 零介入的連續 PWM 串流。

多通道 DMA 連續取樣
c — ADC1 掃描 3 通道（CH0/1/4），DMA 連續存入陣列 
#define ADC_CH_COUNT  3
volatile uint16_t adc_buf[ADC_CH_COUNT];

void adc1_dma_init(void)
{
    /* PA0/PA1/PA4 設為 Analog 模式（MODER = 11）*/
    RCC->AHB1ENR |= RCC_AHB1ENR_GPIOAEN;
    GPIOA->MODER |= (3U<<0) | (3U<<2) | (3U<<8);

    /* ADC 時脈：APB2(90MHz) ÷ 4 = 22.5 MHz */
    RCC->APB2ENR |= RCC_APB2ENR_ADC1EN;
    ADC->CCR = (1U << ADC_CCR_ADCPRE_Pos);   /* ADCPRE = 01 → ÷4 */

    /* DMA2 Stream0 Channel0 = ADC1 */
    RCC->AHB1ENR |= RCC_AHB1ENR_DMA2EN;
    DMA2_Stream0->CR = 0;
    while (DMA2_Stream0->CR & DMA_SxCR_EN);
    DMA2_Stream0->PAR  = (uint32_t)&ADC1->DR;
    DMA2_Stream0->M0AR = (uint32_t)adc_buf;
    DMA2_Stream0->NDTR = ADC_CH_COUNT;
    DMA2_Stream0->CR   =
        (0U << DMA_SxCR_CHSEL_Pos)  |  /* Channel 0 */
        DMA_SxCR_MSIZE_0            |  /* Memory 16-bit */
        DMA_SxCR_PSIZE_0            |  /* Peripheral 16-bit */
        DMA_SxCR_MINC               |  /* Memory 位址遞增 */
        DMA_SxCR_CIRC;                 /* Circular */
    DMA2_Stream0->CR |= DMA_SxCR_EN;

    /* ADC1 設定：12-bit，掃描模式，連續轉換，DMA 請求 */
    ADC1->CR1 = ADC_CR1_SCAN;
    ADC1->CR2 = ADC_CR2_CONT | ADC_CR2_DMA | ADC_CR2_DDS;
    /* 取樣時間：各通道 56 cycles */
    ADC1->SMPR2 = (3U<<0) | (3U<<3) | (3U<<12);   /* CH0 CH1 CH4 */
    /* 序列長度 3，順序 CH0→CH1→CH4 */
    ADC1->SQR1 = ((3-1) << ADC_SQR1_L_Pos);      /* L = 2（3 個通道）*/
    ADC1->SQR3 = (0U<<0) | (1U<<5) | (4U<<10); /* SQ1=CH0, SQ2=CH1, SQ3=CH4 */

    ADC1->CR2 |= ADC_CR2_ADON;   /* 開啟 ADC */
    ADC1->CR2 |= ADC_CR2_SWSTART; /* 啟動連續轉換 */
}

4.6 — 理論 + 實作
CAN Bus
差動訊號、位元時序、仲裁、過濾器配置與錯誤處理
CAN Bus 基礎

CAN（Controller Area Network）採用差動雙絞線（CANH / CANL）， 以顯性（Dominant，低電位差）和隱性（Recessive，高電位差）狀態傳輸， 抗噪能力強，是汽車電子與工業控制的主流匯流排。 標準幀 ID 11-bit，最高 1 Mbps；擴展幀 ID 29-bit。 多節點同時傳輸時，ID 數值較小的節點優先（非破壞性仲裁）。

欄位標準幀（CAN 2.0A）說明
SOF1 bit起始幀（顯性）
ID11 bit優先權 + 識別，數值越小越優先
RTR1 bit0 = 資料幀，1 = 遠端請求幀
DLC4 bit資料長度（0–8 bytes）
Data0–64 bit最多 8 bytes 有效載荷
CRC15 + 1 bitCRC-15 錯誤偵測
ACK2 bit收到者送出顯性 ACK
位元時序與鮑率計算

CAN 位元時間分為四個段：Sync Seg（1 Tq）、 Prop Seg、Phase Seg1、Phase Seg2。 總位元時間 = 各段 Tq 之和。STM32 以 BTR 暫存器設定 BRP、TS1、TS2。

500 kbps 位元時序計算（APB1 45 MHz）
目標：500 kbps，每 bit = 2 µs = 45 MHz / 500 kHz = 90 Tq

BRP = 4（÷5，Tq = 5/45 MHz ≈ 111 ns）
TS1 = 12（13 Tq）、TS2 = 3（4 Tq）
總計：1 + 13 + 4 = 18 Tq，45MHz / (5 × 18) = 500 kHz ✓
取樣點 = (1 + 13) / 18 ≈ 78%（建議範圍 75–90%）
發送與接收 CAN 幀
c — CAN1 初始化、傳送幀與過濾器設定 
void can1_init(void)
{
    RCC->APB1ENR |= RCC_APB1ENR_CAN1EN;
    /* PD0=CAN_RX, PD1=CAN_TX (AF9) — GPIO 設定省略 */

    CAN1->MCR |= CAN_MCR_INRQ;             /* 進入初始化模式 */
    while (!(CAN1->MSR & CAN_MSR_INAK));

    CAN1->MCR &= ~CAN_MCR_SLEEP;
    /* BTR：BRP=4, TS1=12, TS2=3, SJW=0（500 kbps）*/
    CAN1->BTR = (4U  << CAN_BTR_BRP_Pos)  |
                (12U << CAN_BTR_TS1_Pos)  |
                (3U  << CAN_BTR_TS2_Pos);

    /* 過濾器 0：接受所有 ID（遮罩 = 0，過濾器 = 0）*/
    CAN1->FMR  |= CAN_FMR_FINIT;
    CAN1->FA1R &= ~(1U);            /* 停用 Filter 0 */
    CAN1->sFilterRegister[0].FR1 = 0x00000000;
    CAN1->sFilterRegister[0].FR2 = 0x00000000;
    CAN1->FA1R |= (1U);             /* 啟用 Filter 0 */
    CAN1->FMR  &= ~CAN_FMR_FINIT;

    CAN1->MCR &= ~CAN_MCR_INRQ;            /* 離開初始化模式 */
    while (CAN1->MSR & CAN_MSR_INAK);
}

void can1_send(uint32_t id, const uint8_t *data, uint8_t len)
{
    /* 找空的發送信箱 */
    uint8_t mbox = (CAN1->TSR & CAN_TSR_CODE) >> CAN_TSR_CODE_Pos;

    CAN1->sTxMailBox[mbox].TIR  = (id << CAN_TI0R_STID_Pos);
    CAN1->sTxMailBox[mbox].TDTR = len & 0x0F;
    CAN1->sTxMailBox[mbox].TDLR =
        ((uint32_t)data[0])       | ((uint32_t)data[1] << 8)  |
        ((uint32_t)data[2] << 16) | ((uint32_t)data[3] << 24);
    CAN1->sTxMailBox[mbox].TDHR =
        ((uint32_t)data[4])       | ((uint32_t)data[5] << 8)  |
        ((uint32_t)data[6] << 16) | ((uint32_t)data[7] << 24);
    CAN1->sTxMailBox[mbox].TIR |= CAN_TI0R_TXRQ;  /* 請求發送 */
}

本週實作清單
UART DMA 環形緩衝：PC 終端輸入指令，MCU 回應（不阻塞主迴圈）
SPI Flash：讀取 JEDEC ID，寫入 1 頁（256 bytes），讀回驗證
I²C BME280：每秒讀取溫度，透過 UART 輸出「Temp: 25.3°C」
TIM3 PWM：伺服馬達隨搖桿 ADC 值轉動（0°–180°）
ADC DMA 3 通道：計算各通道移動平均值，UART 每 100 ms 輸出一次
WS2812B：以 TIM + DMA 點亮 8 顆 LED，顏色隨 ADC 值漸變
CAN 回環測試（Loopback Mode）：自發自收一筆感測器資料幀，UART 確認
延伸挑戰
以 Idle Line 中斷 + DMA 實作一個 MODBUS RTU Slave： 接收變長度封包，解析功能碼（03 讀保持暫存器、06 寫單一暫存器）， 回應 ADC 原始值與 BME280 感測資料。 這是工業嵌入式系統中最常見的通訊協定之一。
