3.1 — 理論
資料型別、限定詞與記憶體佈局
volatile、const、static、大小端、結構體對齊
volatile — 對編譯器說「不要優化我」

volatile 告訴編譯器：這個變數的值可能在編譯器不知情的情況下改變 （例如硬體暫存器、ISR 寫入的旗標、DMA 目的地緩衝區）。 沒有 volatile，編譯器可能將多次讀取的值快取在暫存器中， 導致程式讀不到硬體的最新狀態。

c — volatile 正確 vs. 錯誤用法 
/* ❌ 錯誤：編譯器可能將迴圈優化為無窮迴圈或直接跳過 */
uint32_t *status_reg = (uint32_t *)0x40020014;
while (*status_reg == 0);   /* 編譯器認為此值不會變，可能優化掉 */

/* ✅ 正確：volatile 強制每次都從記憶體重新讀取 */
volatile uint32_t *status_reg = (volatile uint32_t *)0x40020014;
while (*status_reg == 0);   /* 每次迭代都會真正讀取硬體暫存器 */

/* ISR 與主程式共用旗標，必須 volatile */
volatile uint8_t g_data_ready = 0;

void USART2_IRQHandler(void) {
    g_data_ready = 1;          /* ISR 寫入 */
}

int main(void) {
    while (!g_data_ready);     /* 主迴圈等待：volatile 確保每次重新讀取 */
    process_data();
}
const、static、inline 在嵌入式的含義
限定詞嵌入式場景意義記憶體影響
const 常數資料放在 Flash（唯讀），節省 RAM 存於 .rodata 段（Flash）
static（函式外） 限制連結範圍為本檔案，避免符號衝突 .data 或 .bss（RAM）
static（函式內） 函式呼叫間保留值，但注意執行緒安全 .data 或 .bss（RAM）
inline 提示展開避免呼叫開銷，適合短小的暫存器操作 無固定段（可能重複展開）
__attribute__((section("..."))) 強制指定段名（如向量表放 .isr_vector） 由連結腳本決定
固定寬度整數型別（<stdint.h>）

嵌入式程式應永遠使用固定寬度型別，避免因平台不同造成 int、long 大小差異。

型別寬度有號範圍使用場景
uint8_t8-bit0 – 255位元組緩衝、UART 資料
int8_t8-bit-128 – 127有號感測器原始值
uint16_t16-bit0 – 65535ADC 結果、計時器計數
uint32_t32-bit0 – 4294967295暫存器位址、時間戳
uint64_t64-bit—高精度計時（謹慎使用，無 FPU 時開銷大）
結構體對齊與 Padding

ARM Cortex-M4 對非對齊存取有嚴格限制（預設拋出 HardFault）。 編譯器會在結構體成員間插入 padding 確保對齊，可能造成實際大小遠超預期。

c — 結構體 padding 與 __packed 屬性 
/* sizeof(Sensor) = 12，不是 7！編譯器插入 3 bytes padding */
typedef struct {
    uint8_t  id;        /* offset 0 */
    /* 3 bytes padding */
    uint32_t value;     /* offset 4（4-byte 對齊）*/
    uint16_t flags;     /* offset 8 */
    /* 2 bytes padding（結尾讓整體大小為 4 的倍數）*/
} Sensor;

/* 以小到大排列成員，最小化 padding */
typedef struct {
    uint8_t  id;        /* offset 0 */
    uint8_t  _pad[1];   /* 或乾脆讓編譯器處理 */
    uint16_t flags;     /* offset 2 */
    uint32_t value;     /* offset 4 */
} SensorPacked;      /* sizeof = 8 */

/* __packed：強制無 padding（用於通訊協定解析，有非對齊存取風險）*/
typedef struct __attribute__((packed)) {
    uint8_t  cmd;
    uint32_t payload;
    uint16_t crc;
} UartFrame;         /* sizeof = 7，但存取 payload 可能觸發非對齊中斷 */
__packed 的代價
__packed 結構體在存取非對齊的多位元組成員時， 編譯器會生成多條位元組存取指令來模擬，效能較差。 應只在處理外部協定封包（UART、CAN frame）時使用， 不要用於一般內部資料結構。
3.2 — 實作
位元操作與暫存器慣用法
遮罩、位移、SET/CLEAR/TOGGLE/TEST 四大操作與防禦性寫法
四大位元操作原語
c — 位元操作慣用語（以 bit N 為例） 
/* 假設 reg 是一個 32-bit 暫存器，要操作第 N bit */
#define BIT(n)   (1U << (n))

/* SET：將第 N bit 設為 1，其他 bit 不變 */
reg |= BIT(N);

/* CLEAR：將第 N bit 清為 0，其他 bit 不變 */
reg &= ~BIT(N);

/* TOGGLE：反轉第 N bit */
reg ^= BIT(N);

/* TEST：判斷第 N bit 是否為 1 */
if (reg & BIT(N)) { /* ... */ }

/* 多位元欄位操作（例如 MODER[2N+1:2N]，2-bit 欄位）*/
#define FIELD_MASK(pos, width)  (((1U << (width)) - 1U) << (pos))
#define FIELD_SET(reg, pos, width, val) \
    ((reg) = ((reg) & ~FIELD_MASK(pos, width)) | (((val) << (pos)) & FIELD_MASK(pos, width)))

/* 範例：將 GPIOA_MODER 的 PA5 設為輸出（01）*/
FIELD_SET(GPIOA->MODER, 10, 2, 0x01);   /* pos=10, width=2, val=01 */
以巨集封裝暫存器操作

直接在業務邏輯中散落 GPIOA->BSRR = (1U << 5) 的寫法難以維護。 建立抽象巨集或 inline 函式，讓程式碼表達意圖而非位元操作細節。

c — gpio.h — 可讀性佳的 GPIO 抽象 
#ifndef GPIO_H
#define GPIO_H

#include "stm32f4xx.h"

/* 原子性 SET（使用 BSRR 高 16 bit = Reset，低 16 bit = Set）*/
static inline void gpio_set(GPIO_TypeDef *port, uint16_t pin) {
    port->BSRR = (uint32_t)pin;
}

/* 原子性 RESET */
static inline void gpio_reset(GPIO_TypeDef *port, uint16_t pin) {
    port->BSRR = (uint32_t)pin << 16;
}

/* TOGGLE（用 ODR XOR，非原子，多中斷場合應用 BSRR）*/
static inline void gpio_toggle(GPIO_TypeDef *port, uint16_t pin) {
    port->ODR ^= pin;
}

/* READ */
static inline uint8_t gpio_read(GPIO_TypeDef *port, uint16_t pin) {
    return (port->IDR & pin) ? 1 : 0;
}

/* 腳位定義，對應 STM32F446RE Nucleo */
#define LED_GREEN_PORT   GPIOA
#define LED_GREEN_PIN    (1U << 5)   /* PA5 */
#define BTN_PORT         GPIOC
#define BTN_PIN          (1U << 13)  /* PC13 */

#endif /* GPIO_H */
位元欄位（Bit-field）結構體

C 語言的 bit-field 結構體提供另一種暫存器映射方式，可讀性高但有可攜性問題： 位元填充順序（LSB 或 MSB）由實作定義，不同編譯器可能相反。 CMSIS 標頭檔採用的策略是：以結構體定義命名欄位提升可讀性， 但實際讀寫使用整數型別加位元遮罩確保正確性。

Bit-field 的跨平台陷阱
切勿用 bit-field 結構體直接映射硬體暫存器並依賴欄位順序， 這在更換編譯器或架構時可能靜默失效。 CMSIS 的 _TypeDef 結構體用 __IO uint32_t 整數型別存取暫存器， 再用帶 _Msk、_Pos 後綴的巨集操作，才是正確做法。

指標與記憶體映射

嵌入式 C 中，指標的核心用途之一是將整數位址強轉為可存取的記憶體映射暫存器。 理解指標型別決定的「步進大小」對於位址計算至關重要。

c — 指標算術與記憶體映射 
uint32_t *p = (uint32_t *)0x20000000;
p++;   /* p 現在是 0x20000004（+4 bytes，因為 uint32_t 是 4 bytes）*/

uint8_t *q = (uint8_t *)0x20000000;
q++;   /* q 現在是 0x20000001（+1 byte）*/

/* 走訪 Flash 中的查找表 */
const uint16_t sine_lut[256] = { /* ... */ };
const uint16_t *lut_ptr = sine_lut;
uint16_t sample = lut_ptr[90];   /* 等同 *(lut_ptr + 90) */

/* void 指標：泛型緩衝區（如 memcpy 實作）*/
void my_memcpy(void *dst, const void *src, size_t n) {
    uint8_t       *d = (uint8_t *)dst;
    const uint8_t *s = (const uint8_t *)src;
    while (n--) *d++ = *s++;
}
函式指標與回呼（Callback）

函式指標讓嵌入式驅動程式可以在不修改驅動層的情況下， 將事件通知傳遞給應用層，實現類似物件導向中「介面」的效果。

c — UART 驅動使用回呼解耦應用層 
/* ── uart.h ── */
typedef void (*uart_rx_cb_t)(uint8_t byte);  /* 回呼型別定義 */

void uart_init(uint32_t baud, uart_rx_cb_t cb);
void uart_send(const uint8_t *buf, uint16_t len);

/* ── uart.c ── */
static uart_rx_cb_t s_rx_cb = NULL;  /* 儲存回呼函式指標 */

void uart_init(uint32_t baud, uart_rx_cb_t cb) {
    s_rx_cb = cb;
    /* ... 硬體初始化 ... */
}

void USART2_IRQHandler(void) {
    if (USART2->SR & USART_SR_RXNE) {
        uint8_t data = (uint8_t)(USART2->DR & 0xFF);
        if (s_rx_cb) s_rx_cb(data);    /* 通知應用層，不需知道對方是誰 */
    }
}

/* ── main.c ── */
static void on_byte_received(uint8_t b) {
    /* 應用邏輯：將收到的 byte 存入 ring buffer */
    ring_buf_push(&g_rx_buf, b);
}

int main(void) {
    uart_init(115200, on_byte_received);  /* 傳入回呼 */
    while (1) { /* ... */ }
}
函式指標陣列（狀態機 / 分派表）

函式指標陣列是實作 狀態機 或 命令分派表 的慣用手法， 比大型 switch-case 更具擴充性，且跳轉成本固定為 O(1)。

c — 命令分派表（UART 協定解析） 
typedef void (*cmd_handler_t)(const uint8_t *payload, uint8_t len);

static void cmd_ping   (const uint8_t *p, uint8_t l) { send_pong(); }
static void cmd_led_on (const uint8_t *p, uint8_t l) { gpio_set(LED_GREEN_PORT, LED_GREEN_PIN); }
static void cmd_led_off(const uint8_t *p, uint8_t l) { gpio_reset(LED_GREEN_PORT, LED_GREEN_PIN); }

/* cmd_id 直接作為索引，O(1) 分派 */
static const cmd_handler_t dispatch_table[] = {
    [0x01] = cmd_ping,
    [0x02] = cmd_led_on,
    [0x03] = cmd_led_off,
};
#define CMD_TABLE_SIZE  (sizeof(dispatch_table) / sizeof(dispatch_table[0]))

void dispatch_cmd(uint8_t id, const uint8_t *payload, uint8_t len) {
    if (id < CMD_TABLE_SIZE && dispatch_table[id])
        dispatch_table[id](payload, len);
}
3.4 — 理論 + 實作
連結腳本與啟動碼（Startup）
段（Section）定義、記憶體區域、啟動初始化序列
程式碼段（Sections）一覽
段名存放於內容初始化
.textFlash可執行指令—（直接在 Flash 執行）
.rodataFlashconst 全域變數、字串常數—
.dataFlash（LMA）→ RAM（VMA）有初始值的全域 / static 變數啟動碼將 Flash 資料複製至 RAM
.bssRAM初始值為 0 的全域 / static 變數啟動碼將此段清零
.stackRAM呼叫堆疊MSP 初始值由向量表 [0] 設定
.heapRAM動態記憶體（malloc）嵌入式通常避免使用
最小連結腳本（STM32F446RE）
ld — stm32f446re.ld 
MEMORY {
    FLASH (rx)  : ORIGIN = 0x08000000, LENGTH = 512K
    RAM   (rwx) : ORIGIN = 0x20000000, LENGTH = 128K
}

ENTRY(Reset_Handler)    /* 進入點 */

SECTIONS {
    /* ── 程式碼 ── */
    .text : {
        KEEP(*(.isr_vector))    /* 向量表必須在最前面 */
        *(.text*)               /* 所有 .text 段 */
        *(.rodata*)             /* 唯讀資料 */
        . = ALIGN(4);
    } > FLASH

    /* ── 已初始化資料（Flash → RAM）── */
    .data : {
        _sdata = .;             /* RAM 起始位址（VMA）*/
        *(.data*)
        . = ALIGN(4);
        _edata = .;             /* RAM 結束位址 */
    } > RAM AT > FLASH          /* VMA=RAM, LMA=Flash */
    _sidata = LOADADDR(.data);  /* Flash 中資料的位址（LMA）*/

    /* ── 未初始化資料（清零段）── */
    .bss : {
        _sbss = .;
        *(.bss*)
        *(COMMON)
        . = ALIGN(4);
        _ebss = .;
    } > RAM

    /* ── 堆疊（從 RAM 頂端往下成長）── */
    _estack = ORIGIN(RAM) + LENGTH(RAM);
}
啟動碼：Reset_Handler 做了什麼

在 main() 執行之前，Reset_Handler 必須完成三件事：

1
複製 .data 段：Flash → RAM
使用連結腳本提供的 _sdata、_edata、_sidata 符號， 將 Flash 中的初始值資料搬入 RAM。
c — .data 初始化 
uint32_t *src = &_sidata;
uint32_t *dst = &_sdata;
while (dst < &_edata) *dst++ = *src++;
2
清零 .bss 段
C 語言規定未初始化的全域變數初始值為 0，啟動碼負責執行此操作。
c — .bss 清零 
uint32_t *bss = &_sbss;
while (bss < &_ebss) *bss++ = 0;
3
呼叫 main()
若有 C++ 靜態建構子，需先執行 __libc_init_array()。 完成後跳轉至 main()，main 不應返回；若返回則進入無窮迴圈。

實用 Makefile 範本
makefile — 多檔案 STM32 專案 
TARGET  = blink
MCU     = cortex-m4
FPU     = fpv4-sp-d16
FLOAT   = hard

CC      = arm-none-eabi-gcc
OBJCOPY = arm-none-eabi-objcopy
SIZE    = arm-none-eabi-size

# 來源檔案（自動蒐集）
SRCS    = $(wildcard src/*.c) \
          startup/startup_stm32f446xx.s

OBJS    = $(patsubst %.c, build/%.o, \
          $(patsubst %.s, build/%.o, $(SRCS)))

# 編譯旗標
CFLAGS  = -mcpu=$(MCU) -mthumb \
          -mfpu=$(FPU) -mfloat-abi=$(FLOAT) \
          -O2 -Wall -Wextra \
          -ffunction-sections -fdata-sections \
          -Iinc -DSTM32F446xx

LDFLAGS = -T stm32f446re.ld \
          -Wl,--gc-sections \
          -Wl,-Map=$(TARGET).map \
          --specs=nano.specs \
          -lc -lm

# 預設目標
all: $(TARGET).bin $(TARGET).hex
    $(SIZE) $(TARGET).elf

# 連結
$(TARGET).elf: $(OBJS)
    $(CC) $(CFLAGS) $(LDFLAGS) -o $@ $^

# 編譯 .c
build/%.o: %.c
    @mkdir -p $(dir $@)
    $(CC) $(CFLAGS) -MMD -MP -c $< -o $@

# 組譯 .s
build/%.o: %.s
    @mkdir -p $(dir $@)
    $(CC) $(CFLAGS) -c $< -o $@

# 產生 binary / hex
$(TARGET).bin: $(TARGET).elf
    $(OBJCOPY) -O binary $< $@

$(TARGET).hex: $(TARGET).elf
    $(OBJCOPY) -O ihex $< $@

# 燒錄（st-flash）
flash: $(TARGET).bin
    st-flash write $< 0x08000000

# 燒錄（OpenOCD）
flash-ocd: $(TARGET).elf
    openocd -f interface/stlink.cfg \
            -f target/stm32f4x.cfg \
            -c "program $< verify reset exit"

clean:
    rm -rf build $(TARGET).elf $(TARGET).bin $(TARGET).hex $(TARGET).map

# 自動相依（-MMD -MP 生成 .d 檔）
-include $(OBJS:.o=.d)
重要編譯器旗標說明
旗標作用
-ffunction-sections每個函式獨立放入一個段，讓連結器可以刪除未使用的函式
-fdata-sections同上，用於全域變數
-Wl,--gc-sections連結器刪除未參照的段（Dead Code Elimination）
-Wl,-Map=x.map輸出記憶體映射報告，可查看每個符號的大小與位置
--specs=nano.specs使用 newlib-nano（較小的 C 標準函式庫）
-MMD -MP自動生成標頭檔相依規則（.d 檔），改 .h 後自動重新編譯
-O2二級優化（嵌入式常用）；-Os 優先縮小程式碼大小
-mfloat-abi=hard使用 FPU 硬體浮點（需搭配 -mfpu=fpv4-sp-d16）
解讀 .map 檔案

.map 檔列出每個段的起始位址、大小以及貢獻的目的檔， 是優化 Flash / RAM 用量的必要工具。搭配 arm-none-eabi-size 指令 快速查看 .text、.data、.bss 的總用量。

shell — size 輸出範例 
$ arm-none-eabi-size blink.elf
   text    data     bss     dec     hex filename
   4328      24     104    4456    1168 blink.elf

# text  → Flash 用量（程式碼 + rodata）
# data  → Flash + RAM 各佔（初始值複製）
# bss   → RAM 用量（清零段）
# Flash 總用量 = text + data = 4352 bytes
# RAM   總用量 = data + bss  = 128 bytes

HardFault 分析手法

HardFault 是最常見的嵌入式崩潰。發生時 CPU 將 8 個暫存器 （R0–R3、R12、LR、PC、xPSR）壓入堆疊，並跳轉至 HardFault_Handler。 透過 GDB 讀取堆疊內容，可還原崩潰當下的 PC（造成 Fault 的指令位址）。

c — HardFault Handler（列印暫存器資訊） 
/* 組語部分：判斷使用 MSP 還是 PSP，傳入堆疊指標 */
__attribute__((naked))
void HardFault_Handler(void)
{
    __asm__ volatile (
        "tst lr, #4          \n"   /* 檢查 EXC_RETURN bit 2 */
        "ite eq              \n"
        "mrseq r0, msp       \n"   /* Thread 用 MSP */
        "mrsne r0, psp       \n"   /* Handler 用 PSP */
        "b hard_fault_handler_c \n"
    );
}

void hard_fault_handler_c(uint32_t *stack)
{
    /* 堆疊佈局（由低到高）: R0 R1 R2 R3 R12 LR PC xPSR */
    volatile uint32_t r0   = stack[0];
    volatile uint32_t r1   = stack[1];
    volatile uint32_t r2   = stack[2];
    volatile uint32_t r3   = stack[3];
    volatile uint32_t r12  = stack[4];
    volatile uint32_t lr   = stack[5];
    volatile uint32_t pc   = stack[6];   /* ← 造成 Fault 的指令位址 */
    volatile uint32_t xpsr = stack[7];

    /* 在 GDB 中：(gdb) print/x pc  → 即可反組譯找出問題指令 */
    /* 或查詢 SCB->CFSR（Configurable Fault Status Register）*/
    volatile uint32_t cfsr = SCB->CFSR;  /* 記憶體 / 匯流排 / 使用 Fault 原因 */
    (void)r0; (void)r1; (void)r2; (void)r3;
    (void)r12; (void)lr; (void)xpsr; (void)cfsr;

    while (1);  /* 停在此處，讓 GDB 連入後查看各變數 */
}
addr2line 快速定位
在 GDB 外，可用 arm-none-eabi-addr2line 直接查詢崩潰位址對應的原始碼行數：
arm-none-eabi-addr2line -e blink.elf -f 0x080012A4
