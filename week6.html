6.1 — 理論
開機流程與 Bootloader
ROM → SPL → U-Boot → 核心的四階段開機，U-Boot 環境變數與網路開機
四階段開機序列
上電 / Reset │ ▼ ① ROM Code （片上唯讀，不可修改） 初始化最低限度硬體（SRAM、時脈） 從 SD Card / eMMC / NAND 載入下一階段 │ ▼ ② SPL / MLO （Secondary Program Loader，< 128 KB） 初始化 DRAM 控制器 從儲存媒體載入完整 U-Boot 到 DRAM │ ▼ ③ U-Boot （完整 Bootloader，支援網路 / USB / 環境變數） 設定核心開機參數（bootargs） 載入 zImage + DTB → DRAM 跳轉至核心入口點 │ ▼ ④ Linux Kernel 解壓縮自身、初始化 MMU / 排程器 / 驅動 掛載 rootfs → 執行 /sbin/init（systemd / BusyBox） │ ▼ 使用者空間

設定 bootargs — 核心啟動參數
shell — U-Boot console：從 SD Card 啟動 RPi 4
# 設定核心啟動參數：rootfs 在 mmcblk0p2，使用 ext4 setenv bootargs "console=ttyS0,115200 root=/dev/mmcblk0p2 rootfstype=ext4 rw rootwait" # 載入核心映像與 DTB 到 DRAM fatload mmc 0:1 0x08000000 zImage fatload mmc 0:1 0x08800000 bcm2711-rpi-4-b.dtb # 啟動核心（格式：bootz 核心位址 initrd DTB位址） bootz 0x08000000 - 0x08800000 # 將上述指令存為 bootcmd，開機自動執行 setenv bootcmd "fatload mmc 0:1 0x08000000 zImage; fatload mmc 0:1 0x08800000 bcm2711-rpi-4-b.dtb; bootz 0x08000000 - 0x08800000" saveenv
NFS 根檔案系統（開發利器）

開發期間將 rootfs 掛載在開發主機的 NFS 目錄， 修改檔案後無需重新燒錄 SD Card，重新開機即生效，大幅縮短迭代時間。

shell — U-Boot NFS 開機設定
# 主機端：/etc/exports /opt/rootfs *(rw,sync,no_subtree_check,no_root_squash) # U-Boot bootargs：NFS rootfs setenv bootargs "console=ttyS0,115200 root=/dev/nfs nfsroot=192.168.1.100:/opt/rootfs,v3 ip=192.168.1.50:::::eth0:off rw rootwait" # 透過 TFTP 載入核心（快速迭代，不用換 SD Card） setenv ipaddr 192.168.1.50 setenv serverip 192.168.1.100 tftp 0x08000000 zImage tftp 0x08800000 bcm2711-rpi-4-b.dtb bootz 0x08000000 - 0x08800000
6.2 — 理論 + 實作
Device Tree
DTS 語法、節點 / 屬性 / 相容性字串、Overlay 動態掛載
為什麼需要 Device Tree

ARM 平台的板級硬體配置（哪個 I²C 掛了什麼感測器、哪個 GPIO 接了 LED） 過去以 board file（arch/arm/mach-xxx/board-yyy.c）硬編碼在核心裡， 導致每塊板子都要維護一份核心。 Device Tree 將硬體描述從核心分離出來，同一顆核心可以啟動不同板子， 只需替換 DTB（Device Tree Blob）。

DTS 語法與核心概念
dts — 描述一個掛在 I²C1 的 BME280 感測器
/* DTS 根節點 */ / { /* 相容性字串：核心靠此尋找對應的板級 machine_desc */ compatible = "raspberrypi,4-model-b", "brcm,bcm2711"; #address-cells = <1>; /* reg 屬性中，位址用 1 個 cell 表示 */ #size-cells = <1>; /* reg 屬性中，大小用 1 個 cell 表示 */ /* 記憶體節點 */ memory@0 { device_type = "memory"; reg = <0x00000000 0x40000000>; /* 1 GB，起始 0x0 */ }; /* 參照 SoC 的 I²C1 控制器並新增裝置 */ &i2c1 { status = "okay"; clock-frequency = <400000>; /* 400 kHz 快速模式 */ bme280: bme280@76 { compatible = "bosch,bme280"; /* 驅動程式匹配此字串 */ reg = <0x76>; /* 7-bit I²C 位址 */ /* 可選：中斷腳位描述 */ interrupt-parent = <&gpio>; interrupts = <17 1>; /* GPIO17，上升緣 */ }; }; /* GPIO 控制器引用，設定 LED 腳位 */ leds { compatible = "gpio-leds"; led-green { gpios = <&gpio 27 0>; /* GPIO27，低有效 */ linux,default-trigger = "heartbeat"; default-state = "off"; }; }; };
Device Tree Overlay — 動態掛載周邊

Overlay 允許在不重新編譯完整 DTB 的情況下，動態增加或修改硬體描述。 Raspberry Pi 的 /boot/config.txt 中 dtoverlay= 即使用此機制。

dts — my-bme280-overlay.dts
/* Overlay：在執行期疊加到基礎 DTB */ /dts-v1/; /plugin/; / { compatible = "brcm,bcm2835"; /* 聲明此 overlay 適用的平台 */ fragment@0 { target = <&i2c1>; /* 疊加到 i2c1 節點 */ __overlay__ { #address-cells = <1>; #size-cells = <0>; status = "okay"; bme280@76 { compatible = "bosch,bme280"; reg = <0x76>; }; }; }; };
shell — 編譯並載入 Overlay
# 編譯 .dts → .dtbo dtc -@ -I dts -O dtb -o my-bme280.dtbo my-bme280-overlay.dts # 複製到 overlays 目錄 sudo cp my-bme280.dtbo /boot/overlays/ # 在 /boot/config.txt 加入（RPi 平台） dtoverlay=my-bme280 # 或在執行期動態載入（需要 configfs overlay） sudo dtoverlay my-bme280.dtbo # 確認裝置是否出現 ls /sys/bus/i2c/devices/ i2cdetect -y 1 # 應在位址 0x76 看到裝置

Linux 核心模組
模組生命週期、Kbuild 系統、模組參數、核心日誌與符號匯出
最小核心模組
c — hello.c：最小可載入核心模組
#include <linux/module.h> #include <linux/kernel.h> #include <linux/init.h> /* 模組參數：insmod hello.ko msg="Hi" */ static char *msg = "Hello, kernel!"; module_param(msg, charp, 0644); /* 型別 charp，sysfs 權限 0644 */ MODULE_PARM_DESC(msg, "Message to print on load"); static int __init hello_init(void) { pr_info("hello: loaded — %s\n", msg); return 0; /* 回傳非 0 → 載入失敗，模組不被插入 */ } static void __exit hello_exit(void) { pr_info("hello: unloaded\n"); } module_init(hello_init); module_exit(hello_exit); MODULE_LICENSE("GPL"); /* 必須宣告，否則無法使用 GPL-only 符號 */ MODULE_AUTHOR("Your Name"); MODULE_DESCRIPTION("Minimal kernel module example");
Kbuild Makefile
makefile — 針對目標板交叉編譯
# 指定要編譯的模組（obj-m 告訴 Kbuild 這是一個模組） obj-m += hello.o # 若模組由多個原始檔組成： # mydrv-objs := mydrv_core.o mydrv_i2c.o # obj-m += mydrv.o KDIR ?= /lib/modules/$(shell uname -r)/build ARCH ?= arm64 CROSS ?= aarch64-linux-gnu- all: $(MAKE) -C $(KDIR) M=$(PWD) \ ARCH=$(ARCH) CROSS_COMPILE=$(CROSS) modules clean: $(MAKE) -C $(KDIR) M=$(PWD) clean install: $(MAKE) -C $(KDIR) M=$(PWD) modules_install depmod -a
shell — 模組操作指令
# 插入模組（帶參數） sudo insmod hello.ko msg="Embedded Linux" # 查看核心日誌（即時） dmesg | tail -5 journalctl -k -f # systemd 環境 # 列出已載入模組 lsmod | grep hello # 查看模組資訊（授權、參數、依賴） modinfo hello.ko # 移除模組 sudo rmmod hello # 從 /sys/module 修改參數（若 0644 權限） echo "New message" | sudo tee /sys/module/hello/parameters/msg
核心版本綁定（Module Versioning）
核心模組與特定核心版本緊密耦合。 在 A 版本核心編譯的 .ko 無法在 B 版本核心載入 （除非啟用 CONFIG_MODVERSIONS 且符號 CRC 相符）。 交叉編譯時 KDIR 必須指向與目標板相同版本的核心原始碼樹（或標頭檔）。

字元裝置架構
使用者空間 open("/dev/mydev", O_RDWR) read() / write() / ioctl() │ 系統呼叫 ─────────┼──────────────────────── 核心邊界 │ VFS 層 ──▶ 依主裝置號 (major) 找到 file_operations 表 │ 驅動程式 .open = mydev_open ← 分配 private_data .release = mydev_close ← 釋放資源 .read = mydev_read ← copy_to_user .write = mydev_write ← copy_from_user .unlocked_ioctl = mydev_ioctl ← 控制指令 .mmap = mydev_mmap ← 零拷貝共享記憶體
完整字元裝置驅動實作
c — mydev.c：具備 read/write/ioctl 的字元裝置
#include <linux/module.h> #include <linux/cdev.h> #include <linux/fs.h> #include <linux/uaccess.h> #include <linux/slab.h> #include <linux/device.h> #define DEV_NAME "mydev" #define BUF_SIZE 4096 /* ioctl 命令定義（使用 _IO/_IOR/_IOW 巨集）*/ #define MYDEV_MAGIC 'M' #define MYDEV_RESET _IO(MYDEV_MAGIC, 0) /* 無參數 */ #define MYDEV_GETSTAT _IOR(MYDEV_MAGIC, 1, uint32_t) /* 讀取狀態 */ #define MYDEV_SETFREQ _IOW(MYDEV_MAGIC, 2, uint32_t) /* 寫入參數 */ struct mydev_data { struct cdev cdev; char *buf; size_t buf_len; uint32_t freq_hz; struct mutex lock; }; static struct mydev_data g_dev; static dev_t g_devno; static struct class *g_class; static int mydev_open(struct inode *inode, struct file *filp) { filp->private_data = &g_dev; /* 讓其他 fops 取得裝置上下文 */ pr_debug("mydev: opened\n"); return 0; } static ssize_t mydev_read(struct file *filp, char __user *ubuf, size_t count, loff_t *pos) { struct mydev_data *dev = filp->private_data; ssize_t ret; if (*pos >= dev->buf_len) return 0; /* EOF */ count = min(count, (size_t)(dev->buf_len - *pos)); /* copy_to_user：安全的核心→使用者空間複製 */ ret = copy_to_user(ubuf, dev->buf + *pos, count); if (ret) return -EFAULT; *pos += count; return count; } static ssize_t mydev_write(struct file *filp, const char __user *ubuf, size_t count, loff_t *pos) { struct mydev_data *dev = filp->private_data; count = min(count, (size_t)BUF_SIZE); if (copy_from_user(dev->buf, ubuf, count)) return -EFAULT; dev->buf_len = count; *pos = count; return count; } static long mydev_ioctl(struct file *filp, unsigned int cmd, unsigned long arg) { struct mydev_data *dev = filp->private_data; uint32_t val; switch (cmd) { case MYDEV_RESET: mutex_lock(&dev->lock); memset(dev->buf, 0, BUF_SIZE); dev->buf_len = 0; mutex_unlock(&dev->lock); break; case MYDEV_GETSTAT: val = dev->buf_len; if (copy_to_user((void __user *)arg, &val, sizeof(val))) return -EFAULT; break; case MYDEV_SETFREQ: if (copy_from_user(&val, (void __user *)arg, sizeof(val))) return -EFAULT; dev->freq_hz = val; pr_info("mydev: freq set to %u Hz\n", val); break; default: return -ENOTTY; } return 0; } static const struct file_operations mydev_fops = { .owner = THIS_MODULE, .open = mydev_open, .read = mydev_read, .write = mydev_write, .unlocked_ioctl = mydev_ioctl, }; static int __init mydev_init(void) { int ret; /* 動態申請主裝置號 */ ret = alloc_chrdev_region(&g_devno, 0, 1, DEV_NAME); if (ret) return ret; cdev_init(&g_dev.cdev, &mydev_fops); g_dev.cdev.owner = THIS_MODULE; ret = cdev_add(&g_dev.cdev, g_devno, 1); if (ret) goto err_region; /* 建立 /dev/mydev（需要 udev / mdev）*/ g_class = class_create(DEV_NAME); device_create(g_class, NULL, g_devno, NULL, DEV_NAME); g_dev.buf = kzalloc(BUF_SIZE, GFP_KERNEL); if (!g_dev.buf) { ret = -ENOMEM; goto err_dev; } mutex_init(&g_dev.lock); pr_info("mydev: registered, major=%d\n", MAJOR(g_devno)); return 0; err_dev: device_destroy(g_class, g_devno); class_destroy(g_class); cdev_del(&g_dev.cdev); err_region: unregister_chrdev_region(g_devno, 1); return ret; } static void __exit mydev_exit(void) { kfree(g_dev.buf); device_destroy(g_class, g_devno); class_destroy(g_class); cdev_del(&g_dev.cdev); unregister_chrdev_region(g_devno, 1); } module_init(mydev_init); module_exit(mydev_exit); MODULE_LICENSE("GPL");
核心空間絕不能直接存取使用者空間指標
所有使用者空間 ↔ 核心空間的資料傳輸必須經過 copy_to_user / copy_from_user（或 get_user / put_user）， 這些函式會驗證指標合法性並處理頁面錯誤。 直接解參考 arg 指標可能造成核心崩潰（kernel oops）或安全漏洞。
6.5 — 理論 + 實作
Platform Driver 與 DT 整合
platform_driver / platform_device 模型、of_match_table、資源取得、中斷處理
Platform Driver 框架

Platform Driver 是嵌入式 Linux 驅動最常見的框架，用於驅動「非熱插拔」的 SoC 內建周邊 （GPIO 控制器、UART、SPI、I²C 控制器本身等）。 驅動以 compatible 字串 與 Device Tree 節點綁定， 核心在掃描 DT 時自動完成 probe。

完整 Platform Driver 實作（GPIO LED）
c — gpio-led-driver.c：從 DT 取得 GPIO 並控制 LED
#include <linux/module.h> #include <linux/platform_device.h> #include <linux/of.h> #include <linux/gpio/consumer.h> /* gpiod_ API（推薦）*/ #include <linux/interrupt.h> #include <linux/sysfs.h> struct myled_priv { struct gpio_desc *led_gpiod; /* LED GPIO 描述子 */ struct gpio_desc *btn_gpiod; /* 按鍵 GPIO 描述子 */ int btn_irq; struct device *dev; }; /* sysfs 屬性：/sys/bus/platform/devices/myled.0/led_state */ static ssize_t led_state_store(struct device *dev, struct device_attribute *attr, const char *buf, size_t count) { struct myled_priv *priv = dev_get_drvdata(dev); int val; if (kstrtoint(buf, 10, &val)) return -EINVAL; gpiod_set_value_cansleep(priv->led_gpiod, !!val); return count; } static ssize_t led_state_show(struct device *dev, struct device_attribute *attr, char *buf) { struct myled_priv *priv = dev_get_drvdata(dev); return sysfs_emit(buf, "%d\n", gpiod_get_value(priv->led_gpiod)); } static DEVICE_ATTR_RW(led_state); /* 按鍵中斷處理 */ static irqreturn_t btn_isr(int irq, void *data) { struct myled_priv *priv = data; gpiod_set_value(priv->led_gpiod, !gpiod_get_value(priv->led_gpiod)); return IRQ_HANDLED; } /* probe：核心匹配到 DT 節點時呼叫 */ static int myled_probe(struct platform_device *pdev) { struct myled_priv *priv; int ret; /* devm_ 系列：裝置移除時自動釋放，無需手動 free */ priv = devm_kzalloc(&pdev->dev, sizeof(*priv), GFP_KERNEL); if (!priv) return -ENOMEM; priv->dev = &pdev->dev; /* 從 DT 屬性 "led-gpios" 取得 GPIO 描述子 */ priv->led_gpiod = devm_gpiod_get(&pdev->dev, "led", GPIOD_OUT_LOW); if (IS_ERR(priv->led_gpiod)) return PTR_ERR(priv->led_gpiod); /* 從 DT 屬性 "btn-gpios" 取得按鍵 GPIO */ priv->btn_gpiod = devm_gpiod_get_optional(&pdev->dev, "btn", GPIOD_IN); if (priv->btn_gpiod) { priv->btn_irq = gpiod_to_irq(priv->btn_gpiod); ret = devm_request_irq(&pdev->dev, priv->btn_irq, btn_isr, IRQF_TRIGGER_FALLING, "myled-btn", priv); if (ret) return ret; } platform_set_drvdata(pdev, priv); /* 建立 sysfs 屬性 */ ret = device_create_file(&pdev->dev, &dev_attr_led_state); if (ret) return ret; dev_info(&pdev->dev, "myled: probed successfully\n"); return 0; } static void myled_remove(struct platform_device *pdev) { device_remove_file(&pdev->dev, &dev_attr_led_state); } /* DT 相容性表：與 DTS 中 compatible 屬性匹配 */ static const struct of_device_id myled_of_match[] = { { .compatible = "mycompany,gpio-led-ctrl" }, { } /* 哨兵：必須以空元素結尾 */ }; MODULE_DEVICE_TABLE(of, myled_of_match); static struct platform_driver myled_driver = { .probe = myled_probe, .remove = myled_remove, .driver = { .name = "myled", .of_match_table = myled_of_match, }, }; module_platform_driver(myled_driver); /* 展開 module_init/exit */ MODULE_LICENSE("GPL"); MODULE_DESCRIPTION("GPIO LED Platform Driver with DT and sysfs");
對應的 Device Tree 節點
dts — 與上述驅動配對的 DT 節點
/* 放在板級 DTS 或 Overlay 中 */ myled_ctrl: gpio-led-ctrl { compatible = "mycompany,gpio-led-ctrl"; /* 匹配 of_match_table */ status = "okay"; /* gpiod_get(dev, "led", ...) → 尋找 "led-gpios" 屬性 */ led-gpios = <&gpio 27 GPIO_ACTIVE_HIGH>; btn-gpios = <&gpio 17 GPIO_ACTIVE_LOW>; /* devm_gpiod_get_optional */ };
devm_ 系列 API 的好處
以 devm_ 開頭的資源申請函式（devm_kzalloc、devm_gpiod_get、 devm_request_irq）會將資源綁定到裝置的生命週期。 當 platform_remove 被呼叫時，核心自動按相反順序釋放所有 devm_ 資源， 大幅降低資源洩漏風險，是現代 Linux 驅動的標準寫法。

使用者空間介面與偵錯

從使用者空間操作驅動
c — user_app.c：呼叫 ioctl 控制字元裝置
#include <stdio.h> #include <fcntl.h> #include <unistd.h> #include <sys/ioctl.h> #include <stdint.h> /* 與驅動共享的 ioctl 定義（通常放在共用標頭）*/ #define MYDEV_MAGIC 'M' #define MYDEV_RESET _IO(MYDEV_MAGIC, 0) #define MYDEV_GETSTAT _IOR(MYDEV_MAGIC, 1, uint32_t) #define MYDEV_SETFREQ _IOW(MYDEV_MAGIC, 2, uint32_t) int main(void) { int fd = open("/dev/mydev", O_RDWR); if (fd < 0) { perror("open"); return 1; } /* 重置裝置 */ if (ioctl(fd, MYDEV_RESET) < 0) perror("RESET"); /* 設定頻率 */ uint32_t freq = 1000; if (ioctl(fd, MYDEV_SETFREQ, &freq) < 0) perror("SETFREQ"); /* 寫入資料 */ const char *data = "Hello from userspace"; write(fd, data, strlen(data)); /* 讀取狀態 */ uint32_t stat; if (ioctl(fd, MYDEV_GETSTAT, &stat) == 0) printf("buffer length: %u bytes\n", stat); close(fd); return 0; }
